98 int raspitex_parse_cmdline(RASPITEX_STATE *state,
99  const char *arg1, const char *arg2)
100 {
101  int command_id, used = 0, num_parameters;
102 
103  if (!arg1)
104  return 0;
105 
106  command_id = raspicli_get_command_id(cmdline_commands,
107  cmdline_commands_size, arg1, &num_parameters);
108 
109  // If invalid command, or we are missing a parameter, drop out
110  if (command_id==-1 || (command_id != -1 && num_parameters > 0 && arg2 == NULL))
111  return 0;
112 
113  switch (command_id)
114  {
115  case CommandGLWin: // Allows a GL window to be different to preview-res
116  {
117  int tmp;
118  tmp = sscanf(arg2, "%d,%d,%d,%d",
119  &state->x, &state->y, &state->width, &state->height);
120  if (tmp != 4)
121  {
122  // Default to safe size on parse error
123  state->x = state->y = 0;
124  state->width = DEFAULT_WIDTH;
125  state->height = DEFAULT_HEIGHT;
126  }
127  else
128  {
129  state->gl_win_defined = 1;
130  }
131 
132  used = 2;
133  break;
134  }
135 
136  case CommandGLScene: // Selects the GL scene
137  {
138  if (strcmp(arg2, "square") == 0)
139  state->scene_id = RASPITEX_SCENE_SQUARE;
140  else if (strcmp(arg2, "teapot") == 0)
141  state->scene_id = RASPITEX_SCENE_TEAPOT;
142  else if (strcmp(arg2, "mirror") == 0)
143  state->scene_id = RASPITEX_SCENE_MIRROR;
144  else if (strcmp(arg2, "yuv") == 0)
145  state->scene_id = RASPITEX_SCENE_YUV;
146  else if (strcmp(arg2, "sobel") == 0)
147  state->scene_id = RASPITEX_SCENE_SOBEL;
148  else if (strcmp(arg2, "vcsm_square") == 0)
149  state->scene_id = RASPITEX_SCENE_VCSM_SQUARE;
150  else
151  fprintf(stderr, "Unknown scene %s", arg2);
152 
153  used = 2;
154  break;
155  }
156  }
157  return used;
158 }
