45 // Format for test image
46 const unsigned int WIDTH = 512;
47 const unsigned int HEIGHT = 512;
48 const MMAL_FOURCC_T INPUT_ENC = MMAL_ENCODING_RGBA;
49 const unsigned int BYTESPP = 4;
50 const uint32_t RED = 0xff;
51 const uint32_t GREEN = 0xff << 8;
52 const uint32_t BLUE = 0xff << 16;
53 const uint32_t ALPHA = 0xff << 24;

65 // Create a test input image in the supplied buffer consisting of 8 vertical bars of
66 // white | black | red | green | blue | cyan | magenta | yellow
67 void create_rgba_test_image(void* buf, unsigned int length, unsigned int stride)
68 {
69  uint32_t* pixel = buf;
70  int i;
71  for (i=0; i<length/BYTESPP; ++i) {
72  switch ((i % stride) / (WIDTH / 8)) {
73  case 0: *pixel = RED | GREEN | BLUE; break;
74  case 1: *pixel = 0; break;
75  case 2: *pixel = RED; break;
76  case 3: *pixel = GREEN; break;
77  case 4: *pixel = BLUE; break;
78  case 5: *pixel = GREEN | BLUE; break;
79  case 6: *pixel = RED | BLUE; break;
80  case 7: *pixel = RED | GREEN; break;
81  }
82  *pixel |= ALPHA; // Full alpha
83  ++pixel;
84  }
85 }

87 // mmal_encode_test - Encode a test image and write to file
88 void mmal_encode_test(MMAL_FOURCC_T encoding, // Encoding
89  const char* filename) // File name
90 {
91  MMAL_PORT_T* portIn;
92  MMAL_PORT_T* portOut;
93  MMAL_BUFFER_HEADER_T* in;
94  MMAL_BUFFER_HEADER_T* out;
95  MMAL_STATUS_T status;
96  int eos = 0;
97  int sent = 0;
98  int outputWritten = 0;
99  FILE* outFile;
100  int nw;
101 
102  printf("Encoding test image %s\n", filename);
103 
104  // Configure input
105 
106  portIn = encoder->input[0];
107  encoder->status = MMAL_SUCCESS;
108 
109  if (portIn->is_enabled) {
110  if (mmal_wrapper_port_disable(portIn) != MMAL_SUCCESS) {
111  fprintf(stderr, "Failed to disable input port\n");
112  exit(1);
113  }
114  }
115 
116  portIn->format->encoding = INPUT_ENC;
117  portIn->format->es->video.width = VCOS_ALIGN_UP(WIDTH, 32);
118  portIn->format->es->video.height = VCOS_ALIGN_UP(HEIGHT, 16);
119  portIn->format->es->video.crop.x = 0;
120  portIn->format->es->video.crop.y = 0;
121  portIn->format->es->video.crop.width = WIDTH;
122  portIn->format->es->video.crop.height = HEIGHT;
123  if (mmal_port_format_commit(portIn) != MMAL_SUCCESS) {
124  fprintf(stderr, "Failed to commit input port format\n");
125  exit(1);
126  }
127 
128  portIn->buffer_size = portIn->buffer_size_recommended;
129  portIn->buffer_num = portIn->buffer_num_recommended;
130 
131  if (mmal_wrapper_port_enable(portIn, MMAL_WRAPPER_FLAG_PAYLOAD_ALLOCATE)
132  != MMAL_SUCCESS) {
133  fprintf(stderr, "Failed to enable input port\n");
134  exit(1);
135  }
136 
137  printf("- input %4.4s %ux%u\n",
138  (char*)&portIn->format->encoding,
139  portIn->format->es->video.width, portIn->format->es->video.height);
140 
141  // Configure output
142 
143  portOut = encoder->output[0];
144 
145  if (portOut->is_enabled) {
146  if (mmal_wrapper_port_disable(portOut) != MMAL_SUCCESS) {
147  fprintf(stderr, "Failed to disable output port\n");
148  exit(1);
149  }
150  }
151 
152  portOut->format->encoding = encoding;
153  if (mmal_port_format_commit(portOut) != MMAL_SUCCESS) {
154  fprintf(stderr, "Failed to commit output port format\n");
155  exit(1);
156  }
157 
158  mmal_port_parameter_set_uint32(portOut, MMAL_PARAMETER_JPEG_Q_FACTOR, 100);
159 
160  portOut->buffer_size = portOut->buffer_size_recommended;
161  portOut->buffer_num = portOut->buffer_num_recommended;
162 
163  if (mmal_wrapper_port_enable(portOut, MMAL_WRAPPER_FLAG_PAYLOAD_ALLOCATE)
164  != MMAL_SUCCESS) {
165  fprintf(stderr, "Failed to enable output port\n");
166  exit(1);
167  }
168 
169  printf("- output %4.4s\n", (char*)&encoding);
170 
171  // Perform the encoding
172 
173  outFile = fopen(filename, "w");
174  if (!outFile) {
175  fprintf(stderr, "Failed to open file %s (%s)\n", filename, strerror(errno));
176  exit(1);
177  }
178 
179  while (!eos) {
180 
181  // Send output buffers to be filled with encoded image.
182  while (mmal_wrapper_buffer_get_empty(portOut, &out, 0) == MMAL_SUCCESS) {
183  if (mmal_port_send_buffer(portOut, out) != MMAL_SUCCESS) {
184  fprintf(stderr, "Failed to send buffer\n");
185  break;
186  }
187  }
188 
189  // Send image to be encoded.
190  if (!sent && mmal_wrapper_buffer_get_empty(portIn, &in, 0) == MMAL_SUCCESS) {
191  printf("- sending %u bytes to encoder\n", in->alloc_size);
192  create_rgba_test_image(in->data, in->alloc_size, portIn->format->es->video.width);
193  in->length = in->alloc_size;
194  in->flags = MMAL_BUFFER_HEADER_FLAG_EOS;
195  if (mmal_port_send_buffer(portIn, in) != MMAL_SUCCESS) {
196  fprintf(stderr, "Failed to send buffer\n");
197  break;
198  }
199  sent = 1;
200  }
201 
202  // Get filled output buffers.
203  status = mmal_wrapper_buffer_get_full(portOut, &out, 0);
204  if (status == MMAL_EAGAIN) {
205  // No buffer available, wait for callback and loop.
206  vcos_semaphore_wait(&sem);
207  continue;
208  } else if (status != MMAL_SUCCESS) {
209  fprintf(stderr, "Failed to get full buffer\n");
210  exit(1);
211  }
212 
213  printf("- received %i bytes\n", out->length);
214  eos = out->flags & MMAL_BUFFER_HEADER_FLAG_EOS;
215 
216  nw = fwrite(out->data, 1, out->length, outFile);
217  if (nw != out->length) {
218  fprintf(stderr, "Failed to write complete buffer\n");
219  exit(1);
220  }
221  outputWritten += nw;
222 
223  mmal_buffer_header_release(out);
224  }
225 
226  mmal_port_flush(portOut);
227 
228  fclose(outFile);
229  printf("- written %u bytes to %s\n\n", outputWritten, filename);
230 }
