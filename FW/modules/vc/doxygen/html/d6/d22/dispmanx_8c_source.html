60 static void FillRect( VC_IMAGE_TYPE_T type, void *image, int pitch, int aligned_height, int x, int y, int w, int h, int val )
61 {
62  int row;
63  int col;
64 
65  uint16_t *line = (uint16_t *)image + y * (pitch>>1) + x;
66 
67  for ( row = 0; row < h; row++ )
68  {
69  for ( col = 0; col < w; col++ )
70  {
71  line[col] = val;
72  }
73  line += (pitch>>1);
74  }
75 }
