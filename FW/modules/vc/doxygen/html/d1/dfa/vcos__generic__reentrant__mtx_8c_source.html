44 void vcos_generic_reentrant_mutex_lock(VCOS_REENTRANT_MUTEX_T *m)
45 {
46  VCOS_THREAD_T *thread = vcos_thread_current();
47  vcos_assert(m);
48 
49  vcos_assert(thread != 0);
50 
51  if (m->owner != thread)
52  {
53  vcos_mutex_lock(&m->mutex);
54  m->owner = thread;
55  vcos_assert(m->count == 0);
56  }
57  m->count++;
58 }

60 void vcos_generic_reentrant_mutex_unlock(VCOS_REENTRANT_MUTEX_T *m)
61 {
62  vcos_assert(m->count != 0);
63  vcos_assert(m->owner == vcos_thread_current());

67  m->owner = 0;
