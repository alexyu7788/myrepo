55 portSettingsChanged(OPENMAX_JPEG_DECODER * decoder)
56 {
57  OMX_PARAM_PORTDEFINITIONTYPE portdef;
58 
59  // need to setup the input for the resizer with the output of the
60  // decoder
61  portdef.nSize = sizeof(OMX_PARAM_PORTDEFINITIONTYPE);
62  portdef.nVersion.nVersion = OMX_VERSION;
63  portdef.nPortIndex = decoder->imageDecoder->outPort;
64  OMX_GetParameter(decoder->imageDecoder->handle,
65  OMX_IndexParamPortDefinition, &portdef);
66 
67  unsigned int uWidth =
68  (unsigned int) portdef.format.image.nFrameWidth;
69  unsigned int uHeight =
70  (unsigned int) portdef.format.image.nFrameHeight;
71 
72  // tell resizer input what the decoder output will be providing
73  portdef.nPortIndex = decoder->imageResizer->inPort;
74  OMX_SetParameter(decoder->imageResizer->handle,
75  OMX_IndexParamPortDefinition, &portdef);
76 
77  // establish tunnel between decoder output and resizer input
78  OMX_SetupTunnel(decoder->imageDecoder->handle,
79  decoder->imageDecoder->outPort,
80  decoder->imageResizer->handle,
81  decoder->imageResizer->inPort);
82 
83  // enable ports
84  OMX_SendCommand(decoder->imageDecoder->handle,
85  OMX_CommandPortEnable,
86  decoder->imageDecoder->outPort, NULL);
87  OMX_SendCommand(decoder->imageResizer->handle,
88  OMX_CommandPortEnable,
89  decoder->imageResizer->inPort, NULL);
90 
91  // put resizer in idle state (this allows the outport of the decoder
92  // to become enabled)
93  OMX_SendCommand(decoder->imageResizer->handle,
94  OMX_CommandStateSet, OMX_StateIdle, NULL);
95 
96  // wait for state change complete
97  ilclient_wait_for_event(decoder->imageResizer->component,
98  OMX_EventCmdComplete,
99  OMX_CommandStateSet, 1,
100  OMX_StateIdle, 1, 0, TIMEOUT_MS);
101 
102  // once the state changes, both ports should become enabled and the
103  // resizer
104  // output should generate a settings changed event
105  ilclient_wait_for_event(decoder->imageDecoder->component,
106  OMX_EventCmdComplete,
107  OMX_CommandPortEnable, 1,
108  decoder->imageDecoder->outPort, 1, 0,
109  TIMEOUT_MS);
110  ilclient_wait_for_event(decoder->imageResizer->component,
111  OMX_EventCmdComplete, OMX_CommandPortEnable, 1,
112  decoder->imageResizer->inPort, 1, 0,
113  TIMEOUT_MS);
114  ilclient_wait_for_event(decoder->imageResizer->component,
115  OMX_EventPortSettingsChanged,
116  decoder->imageResizer->outPort, 1, 0, 1, 0,
117  TIMEOUT_MS);
118 
119  ilclient_disable_port(decoder->imageResizer->component,
120  decoder->imageResizer->outPort);
121 
122  // query output buffer requirements for resizer
123  portdef.nSize = sizeof(OMX_PARAM_PORTDEFINITIONTYPE);
124  portdef.nVersion.nVersion = OMX_VERSION;
125  portdef.nPortIndex = decoder->imageResizer->outPort;
126  OMX_GetParameter(decoder->imageResizer->handle,
127  OMX_IndexParamPortDefinition, &portdef);
128 
129  // change output color format and dimensions to match input
130  portdef.format.image.eCompressionFormat = OMX_IMAGE_CodingUnused;
131  portdef.format.image.eColorFormat = OMX_COLOR_Format32bitABGR8888;
132  portdef.format.image.nFrameWidth = uWidth;
133  portdef.format.image.nFrameHeight = uHeight;
134  portdef.format.image.nStride = 0;
135  portdef.format.image.nSliceHeight = 0;
136  portdef.format.image.bFlagErrorConcealment = OMX_FALSE;
137 
138  OMX_SetParameter(decoder->imageResizer->handle,
139  OMX_IndexParamPortDefinition, &portdef);
140 
141  // grab output requirements again to get actual buffer size
142  // requirement (and buffer count requirement!)
143  OMX_GetParameter(decoder->imageResizer->handle,
144  OMX_IndexParamPortDefinition, &portdef);
145 
146  // move resizer into executing state
147  ilclient_change_component_state(decoder->imageResizer->component,
148  OMX_StateExecuting);
149 
150  // show some logging so user knows it's working
151  printf
152  ("Width: %u Height: %u Output Color Format: 0x%x Buffer Size: %u\n",
153  (unsigned int) portdef.format.image.nFrameWidth,
154  (unsigned int) portdef.format.image.nFrameHeight,
155  (unsigned int) portdef.format.image.eColorFormat,
156  (unsigned int) portdef.nBufferSize);
157  fflush(stdout);
158 
159  // enable output port of resizer
160  OMX_SendCommand(decoder->imageResizer->handle,
161  OMX_CommandPortEnable,
162  decoder->imageResizer->outPort, NULL);
163 
164  // allocate the buffer
165  // void* outputBuffer = 0;
166  // if (posix_memalign(&outputBuffer, portdef.nBufferAlignment,
167  // portdef.nBufferSize) != 0)
168  // {
169  // perror("Allocating output buffer");
170  // return OMXJPEG_ERROR_MEMORY;
171  // }
172 
173  // set the buffer

246 
247  int ret = ilclient_create_component(decoder->client,
248  &decoder->
249  imageResizer->

253  |
254  ILCLIENT_ENABLE_INPUT_BUFFERS
255  |

257  if (ret != 0) {

259  return OMXJPEG_ERROR_CREATING_COMP;

273  return OMXJPEG_ERROR_WRONG_NO_PORTS;

276  decoder->imageResizer->outPort = port.nStartPortNumber + 1;
277 
278  decoder->pOutputBufferHeader = NULL;
279 
280  return OMXJPEG_OK;

325 int
326 startupImageDecoder(OPENMAX_JPEG_DECODER * decoder)
327 {
328  // move to idle
329  ilclient_change_component_state(decoder->imageDecoder->component,
330  OMX_StateIdle);
331 
332  // set input image format
333  OMX_IMAGE_PARAM_PORTFORMATTYPE imagePortFormat;
334  memset(&imagePortFormat, 0, sizeof(OMX_IMAGE_PARAM_PORTFORMATTYPE));
335  imagePortFormat.nSize = sizeof(OMX_IMAGE_PARAM_PORTFORMATTYPE);
336  imagePortFormat.nVersion.nVersion = OMX_VERSION;
337  imagePortFormat.nPortIndex = decoder->imageDecoder->inPort;
338  imagePortFormat.eCompressionFormat = OMX_IMAGE_CodingJPEG;
339  OMX_SetParameter(decoder->imageDecoder->handle,
340  OMX_IndexParamImagePortFormat, &imagePortFormat);
341 
342  // get buffer requirements
343  OMX_PARAM_PORTDEFINITIONTYPE portdef;
344  portdef.nSize = sizeof(OMX_PARAM_PORTDEFINITIONTYPE);
345  portdef.nVersion.nVersion = OMX_VERSION;
346  portdef.nPortIndex = decoder->imageDecoder->inPort;
347  OMX_GetParameter(decoder->imageDecoder->handle,
348  OMX_IndexParamPortDefinition, &portdef);
349 
350  // enable the port and setup the buffers
351  OMX_SendCommand(decoder->imageDecoder->handle,
352  OMX_CommandPortEnable,
353  decoder->imageDecoder->inPort, NULL);
354  decoder->inputBufferHeaderCount = portdef.nBufferCountActual;
355  // allocate pointer array
356  decoder->ppInputBufferHeader =
357  (OMX_BUFFERHEADERTYPE **) malloc(sizeof(void) *
358  decoder->inputBufferHeaderCount);
359  // allocate each buffer
360  int i;
361  for (i = 0; i < decoder->inputBufferHeaderCount; i++) {
362  if (OMX_AllocateBuffer(decoder->imageDecoder->handle,
363  &decoder->ppInputBufferHeader[i],
364  decoder->imageDecoder->inPort,
365  (void *) i,
366  portdef.nBufferSize) != OMX_ErrorNone) {
367  perror("Allocate decode buffer");
368  return OMXJPEG_ERROR_MEMORY;
369  }
370  }
371  // wait for port enable to complete - which it should once buffers are
372  // assigned
373  int ret =
374  ilclient_wait_for_event(decoder->imageDecoder->component,
375  OMX_EventCmdComplete,
376  OMX_CommandPortEnable, 0,
377  decoder->imageDecoder->inPort, 0,
378  0, TIMEOUT_MS);
379  if (ret != 0) {
380  fprintf(stderr, "Did not get port enable %d\n", ret);
381  return OMXJPEG_ERROR_EXECUTING;
382  }
383  // start executing the decoder
384  ret = OMX_SendCommand(decoder->imageDecoder->handle,
385  OMX_CommandStateSet, OMX_StateExecuting, NULL);
386  if (ret != 0) {
387  fprintf(stderr, "Error starting image decoder %x\n", ret);
388  return OMXJPEG_ERROR_EXECUTING;
389  }
390  ret = ilclient_wait_for_event(decoder->imageDecoder->component,
391  OMX_EventCmdComplete,
392  OMX_StateExecuting, 0, 0, 1, 0,
393  TIMEOUT_MS);
394  if (ret != 0) {
395  fprintf(stderr, "Did not receive executing stat %d\n", ret);
396  // return OMXJPEG_ERROR_EXECUTING;
397  }
398 
399  return OMXJPEG_OK;
400 }

423  // prepare the image decoder

425  if (ret != OMXJPEG_OK)
426  return ret;
427 
428  ret = prepareResizer(pDecoder[0]);
429  if (ret != OMXJPEG_OK)
430  return ret;
431 
432  ret = startupImageDecoder(pDecoder[0]);
433  if (ret != OMXJPEG_OK)
434  return ret;
435 
436  return OMXJPEG_OK;

439 // this function passed the jpeg image buffer in, and returns the decoded
440 // image
441 int
442 decodeImage(OPENMAX_JPEG_DECODER * decoder, char *sourceImage,
443  size_t imageSize)
444 {
445  char *sourceOffset = sourceImage; // we store a separate
446  // buffer ot image so we
447  // can offset it
448  size_t toread = 0; // bytes left to read from buffer
449  toread += imageSize;
450  int bFilled = 0; // have we filled our output
451  // buffer
452  bufferIndex = 0;
453 
454  while (toread > 0) {
455  // get next buffer from array
456  OMX_BUFFERHEADERTYPE *pBufHeader =
457  decoder->ppInputBufferHeader[bufferIndex];
458 
459  // step index and reset to 0 if required
460  bufferIndex++;
461  if (bufferIndex >= decoder->inputBufferHeaderCount)
462  bufferIndex = 0;
463 
464  // work out the next chunk to load into the decoder
465  if (toread > pBufHeader->nAllocLen)
466  pBufHeader->nFilledLen = pBufHeader->nAllocLen;
467  else
468  pBufHeader->nFilledLen = toread;
469 
470  toread = toread - pBufHeader->nFilledLen;
471 
472  // pass the bytes to the buffer
473  memcpy(pBufHeader->pBuffer, sourceOffset, pBufHeader->nFilledLen);
474 
475  // update the buffer pointer and set the input flags
476 
477  sourceOffset = sourceOffset + pBufHeader->nFilledLen;
478  pBufHeader->nOffset = 0;
479  pBufHeader->nFlags = 0;
480  if (toread <= 0) {
481  pBufHeader->nFlags = OMX_BUFFERFLAG_EOS;
482  }
483  // empty the current buffer
484  int ret =
485  OMX_EmptyThisBuffer(decoder->imageDecoder->handle,
486  pBufHeader);
487 
488  if (ret != OMX_ErrorNone) {
489  perror("Empty input buffer");
490  fprintf(stderr, "return code %x\n", ret);
491  return OMXJPEG_ERROR_MEMORY;
492  }
493  // wait for buffer to empty or port changed event
494  int done = 0;
495  while ((done == 0) && (decoder->pOutputBufferHeader == NULL)) {
496  if (decoder->pOutputBufferHeader == NULL) {
497  ret =
498  ilclient_wait_for_event
499  (decoder->imageDecoder->component,
500  OMX_EventPortSettingsChanged,
501  decoder->imageDecoder->outPort, 0, 0, 1, 0, 5);
502 
503  if (ret == 0) {
504  ret = portSettingsChanged(decoder);
505  if (ret != OMXJPEG_OK)
506  return ret;
507  }
508  } else {
509  ret =
510  ilclient_remove_event(decoder->imageDecoder->component,
511  OMX_EventPortSettingsChanged,
512  decoder->imageDecoder->outPort,
513  0, 0, 1);
514  if (ret == 0)
515  portSettingsChangedAgain(decoder);
516 
517  }
518 
519  // check to see if buffer is now empty
520  if (pBufHeader->nFilledLen == 0)
521  done = 1;
522 
523  if ((done == 0)
524  || (decoder->pOutputBufferHeader == NULL))
525  sleep(1);
526  }
527 
528  // fill the buffer if we have created the buffer
529  if ((bFilled == 0) && (decoder->pOutputBufferHeader != NULL)) {
530  ret = OMX_FillThisBuffer(decoder->imageResizer->handle,
531  decoder->pOutputBufferHeader);
532  if (ret != OMX_ErrorNone) {
533  perror("Filling output buffer");
534  fprintf(stderr, "Error code %x\n", ret);
535  return OMXJPEG_ERROR_MEMORY;
536  }
537 
538  bFilled = 1;
539  }
540  }
541 
542  // wait for buffer to fill
543  /*
544  * while(pBufHeader->nFilledLen == 0) { sleep(5); }
545  */
546 
547  // wait for end of stream events
548  int ret =
549  ilclient_wait_for_event(decoder->imageDecoder->component,
550  OMX_EventBufferFlag,
551  decoder->imageDecoder->outPort, 1,
552  OMX_BUFFERFLAG_EOS, 1,
553  0, 2);
554  if (ret != 0) {
555  fprintf(stderr, "No EOS event on image decoder %d\n", ret);
556  }
557  ret = ilclient_wait_for_event(decoder->imageResizer->component,
558  OMX_EventBufferFlag,
559  decoder->imageResizer->outPort, 1,
560  OMX_BUFFERFLAG_EOS, 1, 0, 2);
561  if (ret != 0) {
562  fprintf(stderr, "No EOS event on image resizer %d\n", ret);
563  }
564  return OMXJPEG_OK;
565 }
