45 /*
46  Binding flags
47  These specify which, if any, of the 3 EGL binding mechanisms this image is
48  taking part in. This information is stored in the image itself rather than
49  the API objects which use it, to enable communication between objects.
50  (For example if a pbuffer is bound to a texture, releasing the texture means
51  the pbuffer can now be bound to a different texture).
52  The IMAGE_FLAG_BOUND_CLIENTBUFFER and IMAGE_FLAG_BOUND_TEXIMAGE flags may be
53  used together, but not with IMAGE_FLAG_BOUND_EGLIMAGE (giving a total of 5
54  possible states).
55 
56  texture is responsible for setting and clearing IMAGE_FLAG_BOUND_TEXIMAGE.
57  EGL is responsible for setting and clearing IMAGE_FLAG_BOUND_CLIENTBUFFER.
58  EGL is responsible for setting IMAGE_FLAG_BOUND_EGLIMAGE (this flag never
59  gets cleared once set).
60 */
61 #define IMAGE_FLAG_BOUND_CLIENTBUFFER (1<<0)
62 #define IMAGE_FLAG_BOUND_TEXIMAGE (1<<1)
63 #define IMAGE_FLAG_BOUND_EGLIMAGE (1<<2)
64 
65 /* if these flags are set, the image should be usable in that way (eg if the
66  * texture flag is set it should be ok to use this image as a texture). the
67  * opposite may not be true. khrn_image_create_dup preserves these flags (so if
68  * you duplicate an image that can be used as a texture, the new image will also
69  * be usable as a texture) */
70 #define IMAGE_FLAG_RENDER_TARGET (1<<3)
71 #define IMAGE_FLAG_TEXTURE (1<<4)
72 #define IMAGE_FLAG_RSO_TEXTURE (1<<5)
73 #define IMAGE_FLAG_DISPLAY (1<<6)

199 /* vc_image->image_data will be set to NULL */
200 static INLINE void khrn_image_fill_vcimage(const KHRN_IMAGE_T *image, VC_IMAGE_T *vc_image)
201 {
202  memset(vc_image, 0, sizeof(*vc_image));
203 
204  if (khrn_image_is_color(image->format)) {
205  switch (image->format & ~(IMAGE_FORMAT_PRE | IMAGE_FORMAT_LIN)) {
206  case ABGR_8888_TF: vc_image->type = VC_IMAGE_TF_RGBA32; break;
207  case XBGR_8888_TF: vc_image->type = VC_IMAGE_TF_RGBX32; break;
208  case RGBA_4444_TF: vc_image->type = VC_IMAGE_TF_RGBA16; break;
209  case RGBA_5551_TF: vc_image->type = VC_IMAGE_TF_RGBA5551; break;
210  case RGB_565_TF: vc_image->type = VC_IMAGE_TF_RGB565; break;
211  case RGBA_8888_RSO:
212  case ABGR_8888_RSO:
213  case ARGB_8888_RSO: vc_image->type = VC_IMAGE_RGBA32; break; //TODO: color channels in the right order? Right upside-downness? (one of these is wrong!)
214  case RGBX_8888_RSO: vc_image->type = VC_IMAGE_RGBX32; break;
215  case XBGR_8888_RSO: vc_image->type = VC_IMAGE_RGBX32; break; //TODO: color channels in the right order? Right upside-downness? (one of these is wrong!)
216  case RGB_565_RSO: vc_image->type = VC_IMAGE_RGB565; break; //if you change these, make sure it doesn't break EGL_KHR_lock_surface
217  case ARGB_4444_RSO: vc_image->type = VC_IMAGE_RGBA16; break;
218  default:
219  UNREACHABLE();
220  vc_image->type = 0;
221  }
222  } else {
223  UNREACHABLE();
224  vc_image->type = 0;
225  }
226  vc_image->width = image->width;
227  vc_image->height = image->height;
228  vc_image->pitch = image->stride;
229  vc_image->size = mem_get_size(image->mh_storage);
230  vc_image->mem_handle = image->mh_storage;
231 }
