50 static unsigned long reverse_byte_order( unsigned long value )
51 {
52  /* Reverse the order of the bytes in the word */
53  return ((value << 24) | ((value & 0xFF00) << 8) | ((value >> 8) & 0xFF00) | (value >> 24));
