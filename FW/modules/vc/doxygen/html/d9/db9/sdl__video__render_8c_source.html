64

104 static MMAL_STATUS_T mmal_sdl_create_surface(MMAL_COMPONENT_MODULE_T *module)
105 {
106  uint32_t flags;
107  int bpp;
108  int w = module->display_region.width;
109  int h = module->display_region.height;
110 
111  flags = SDL_ANYFORMAT | SDL_HWPALETTE | SDL_HWSURFACE | SDL_DOUBLEBUF | SDL_RESIZABLE;
112  bpp = SDL_VideoModeOK(w, h, 16, flags);
113  if(!bpp)
114  {
115  LOG_ERROR("no SDL video mode available");
116  return MMAL_ENOSYS;
117  }
118  module->sdl_surface = SDL_SetVideoMode(w, h, bpp, flags);
119  if(!module->sdl_surface)
120  {
121  LOG_ERROR("cannot create SDL surface");
122  return MMAL_ENOMEM;
123  }
124  return MMAL_SUCCESS;
125 }

129 static MMAL_STATUS_T sdl_port_set_format(MMAL_PORT_T *port)
130 {
131  MMAL_COMPONENT_T *component = port->component;
132  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
133  MMAL_STATUS_T status;
134 
135  if ((status=mmal_sdl_create_surface(module)) != MMAL_SUCCESS)
136  return status;
137 
138  /* We only support I420 */
139  if (port->format->encoding != MMAL_ENCODING_I420)
140  return MMAL_ENOSYS;
141 
142  /* Check if we need to re-create an overlay */
143  if (module->sdl_overlay &&
144  module->width == port->format->es->video.width &&
145  module->height == port->format->es->video.height)
146  return MMAL_SUCCESS; /* Nothing to do */
147 
148  if (module->sdl_overlay)
149  SDL_FreeYUVOverlay(module->sdl_overlay);
150 
151  /* Create overlay */
152  module->sdl_overlay =
153  SDL_CreateYUVOverlay(port->format->es->video.width,
154  port->format->es->video.height,
155  SDL_YV12_OVERLAY, module->sdl_surface);
156  if (!module->sdl_overlay)
157  {
158  LOG_ERROR("cannot create SDL overlay");
159  return MMAL_ENOSPC;
160  }
161  module->width = port->format->es->video.width;
162  module->height = port->format->es->video.height;
163 
164  port->buffer_size_min = module->width * module->height * 3 / 2;
165  return MMAL_SUCCESS;
166 }

184 static MMAL_STATUS_T sdl_port_flush(MMAL_PORT_T *port)
185 {
186  MMAL_COMPONENT_T *component = port->component;
187  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
188  MMAL_BUFFER_HEADER_T *buffer;
189 
190  /* Flush buffers that our component is holding on to.
191  * If the reading thread is holding onto a buffer it will send it back ASAP as well
192  * so no need to care about that. */
193  while((buffer = mmal_queue_get(module->queue)))
194  mmal_port_buffer_header_callback(port, buffer);
195 
196  return MMAL_SUCCESS;

199 static MMAL_BOOL_T sdl_do_processing(MMAL_COMPONENT_T *component)
200 {
201  MMAL_PORT_T *port = component->input[0];
202  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
203  unsigned int width = port->format->es->video.width;
204  unsigned int height = port->format->es->video.height;
205  MMAL_BUFFER_HEADER_T *buffer;
206  uint8_t *src_plane[3];
207  uint32_t *src_pitch;
208  unsigned int i, line;
209  MMAL_BOOL_T eos;
210  SDL_Rect rect;
211 
212  buffer = mmal_queue_get(module->queue);
213  if (!buffer)
214  return 0;
215 
216  eos = buffer->flags & MMAL_BUFFER_HEADER_FLAG_EOS;
217 
218  /* Handle event buffers */
219  if (buffer->cmd)
220  {
221  MMAL_EVENT_FORMAT_CHANGED_T *event = mmal_event_format_changed_get(buffer);
222  if (event)
223  {
224  mmal_format_copy(port->format, event->format);
225  module->status = port->priv->pf_set_format(port);
226  if (module->status != MMAL_SUCCESS)
227  {
228  LOG_ERROR("format not set on port %p", port);
229  if (mmal_event_error_send(port->component, module->status) != MMAL_SUCCESS)
230  LOG_ERROR("unable to send an error event buffer");
231  }
232  }
233  else
234  {
235  LOG_ERROR("discarding event %i on port %p", (int)buffer->cmd, port);
236  }
237 
238  buffer->length = 0;
239  mmal_port_buffer_header_callback(port, buffer);
240  return 1;
241  }
242 
243  if (module->status != MMAL_SUCCESS)
244  return 1;
245 
246  /* Ignore empty buffers */
247  if (!buffer->length)
248  goto end;
249 
250  // FIXME: sanity check the size of the buffer
251 
252  /* Blit the buffer onto the overlay. */
253  src_pitch = buffer->type->video.pitch;
254  src_plane[0] = buffer->data + buffer->type->video.offset[0];
255  src_plane[1] = buffer->data + buffer->type->video.offset[2];
256  src_plane[2] = buffer->data + buffer->type->video.offset[1];
257 
258  SDL_LockYUVOverlay(module->sdl_overlay);
259  for (i=0; i<3; i++)
260  {
261  uint8_t *src = src_plane[i];
262  uint8_t *dst = module->sdl_overlay->pixels[i];
263 
264  if(i == 1) {width /= 2; height /= 2;}
265  for(line = 0; line < height; line++)
266  {
267  memcpy(dst, src, width);
268  src += src_pitch[i];
269  dst += module->sdl_overlay->pitches[i];
270  }
271  }
272  SDL_UnlockYUVOverlay(module->sdl_overlay);
273 
274  width = port->format->es->video.width;
275  height = port->format->es->video.height;
276  rect.x = module->display_region.x;
277  rect.w = module->display_region.width;
278  height = rect.w * height / width;
279  rect.y = module->display_region.y + (module->display_region.height - height) / 2;
280  rect.h = height;
281 
282  SDL_DisplayYUVOverlay(module->sdl_overlay, &rect);
283 
284  end:
285  buffer->offset = buffer->length = 0;
286  mmal_port_buffer_header_callback(port, buffer);
287 
288  /* Generate EOS events */
289  if (eos)
290  mmal_event_eos_send(port);
291 
292  return 1;
293 }

314 static int sdl_event_thread(void *arg)
315 {
316  MMAL_COMPONENT_T *component = (MMAL_COMPONENT_T *)arg;
317  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
318  SDL_Event event;
319 
320  while (SDL_WaitEvent(&event))
321  {
322  switch (event.type)
323  {
324  case SDL_QUIT:
325  if (!module->quit)
326  mmal_event_error_send(component, MMAL_SUCCESS);
327  return 0;
328  default:
329  break;
330  }
331  }
332 
333  return 0;
334 }

337 static MMAL_STATUS_T mmal_component_create_sdl(const char *name, MMAL_COMPONENT_T *component)
338 {
339  MMAL_COMPONENT_MODULE_T *module;
340  MMAL_STATUS_T status = MMAL_ENOMEM;
341 
342  /* Check we're the requested component */
343  if(strcmp(name, "sdl." MMAL_VIDEO_RENDER))
344  return MMAL_ENOENT;
345 
346  if(SDL_Init(SDL_INIT_VIDEO|SDL_INIT_EVENTTHREAD|SDL_INIT_NOPARACHUTE) < 0)
347  return MMAL_ENXIO;
348 
349  /* Allocate our module context */
350  component->priv->module = module = vcos_calloc(1, sizeof(*module), "mmal module");
351  if(!module) return MMAL_ENOMEM;
352 
353  module->queue = mmal_queue_create();
354  if(!module->queue) goto error;
355 
356  /* Allocate the ports for this component */
357  component->input = mmal_ports_alloc(component, 1, MMAL_PORT_TYPE_INPUT, 0);
358  if(!component->input) goto error;
359  component->input_num = 1;
360  module->display_region.width = SDL_WIDTH;
361  module->display_region.height = SDL_HEIGHT;
362 
363  /************/
364 
365  component->input[0]->priv->pf_set_format = sdl_port_set_format;
366  component->input[0]->priv->pf_enable = sdl_port_enable;
367  component->input[0]->priv->pf_disable = sdl_port_disable;
368  component->input[0]->priv->pf_flush = sdl_port_flush;
369  component->input[0]->priv->pf_send = sdl_port_send;
370  component->input[0]->priv->pf_parameter_set = sdl_port_parameter_set;
371  component->input[0]->buffer_num_min = INPUT_MIN_BUFFER_NUM;
372  component->input[0]->buffer_num_recommended = INPUT_RECOMMENDED_BUFFER_NUM;
373 
374  component->priv->pf_destroy = sdl_component_destroy;
375 
376  /* Create a thread to monitor SDL events */
377  module->thread = SDL_CreateThread(sdl_event_thread, component);
378 
379  status = mmal_component_action_register(component, sdl_do_processing_loop);
380  if (status != MMAL_SUCCESS)
381  goto error;
382 
383  return MMAL_SUCCESS;
384 
385  error:
386  sdl_component_destroy(component);
387  return status;
388 }
