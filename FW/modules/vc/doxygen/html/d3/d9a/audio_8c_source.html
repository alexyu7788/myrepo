71 int32_t audioplay_create(AUDIOPLAY_STATE_T **handle,
72  uint32_t sample_rate,
73  uint32_t num_channels,
74  uint32_t bit_depth,
75  uint32_t num_buffers,
76  uint32_t buffer_size)
77 {
78  uint32_t bytes_per_sample = (bit_depth * OUT_CHANNELS(num_channels)) >> 3;
79  int32_t ret = -1;
80 
81  *handle = NULL;
82 
83  // basic sanity check on arguments
84  if(sample_rate >= 8000 && sample_rate <= 192000 &&
85  (num_channels >= 1 && num_channels <= 8) &&
86  (bit_depth == 16 || bit_depth == 32) &&
87  num_buffers > 0 &&
88  buffer_size >= bytes_per_sample)
89  {
90  // buffer lengths must be 16 byte aligned for VCHI
91  int size = (buffer_size + 15) & ~15;
92  AUDIOPLAY_STATE_T *st;
93 
94  // buffer offsets must also be 16 byte aligned for VCHI
95  st = calloc(1, sizeof(AUDIOPLAY_STATE_T));
96 
97  if(st)
98  {
99  OMX_ERRORTYPE error;
100  OMX_PARAM_PORTDEFINITIONTYPE param;
101  OMX_AUDIO_PARAM_PCMMODETYPE pcm;
102  int32_t s;
103 
104  ret = 0;
105  *handle = st;
106 
107  // create and start up everything
108  s = sem_init(&st->sema, 0, 1);
109  assert(s == 0);
110 
111  st->bytes_per_sample = bytes_per_sample;
112  st->num_buffers = num_buffers;
113 
114  st->client = ilclient_init();
115  assert(st->client != NULL);
116 
117  ilclient_set_empty_buffer_done_callback(st->client, input_buffer_callback, st);
118 
119  error = OMX_Init();
120  assert(error == OMX_ErrorNone);
121 
122  ilclient_create_component(st->client, &st->audio_render, "audio_render", ILCLIENT_ENABLE_INPUT_BUFFERS | ILCLIENT_DISABLE_ALL_PORTS);
123  assert(st->audio_render != NULL);
124 
125  st->list[0] = st->audio_render;
126 
127  // set up the number/size of buffers
128  memset(&param, 0, sizeof(OMX_PARAM_PORTDEFINITIONTYPE));
129  param.nSize = sizeof(OMX_PARAM_PORTDEFINITIONTYPE);
130  param.nVersion.nVersion = OMX_VERSION;
131  param.nPortIndex = 100;
132 
133  error = OMX_GetParameter(ILC_GET_HANDLE(st->audio_render), OMX_IndexParamPortDefinition, &param);
134  assert(error == OMX_ErrorNone);
135 
136  param.nBufferSize = size;
137  param.nBufferCountActual = num_buffers;
138 
139  error = OMX_SetParameter(ILC_GET_HANDLE(st->audio_render), OMX_IndexParamPortDefinition, &param);
140  assert(error == OMX_ErrorNone);
141 
142  // set the pcm parameters
143  memset(&pcm, 0, sizeof(OMX_AUDIO_PARAM_PCMMODETYPE));
144  pcm.nSize = sizeof(OMX_AUDIO_PARAM_PCMMODETYPE);
145  pcm.nVersion.nVersion = OMX_VERSION;
146  pcm.nPortIndex = 100;
147  pcm.nChannels = OUT_CHANNELS(num_channels);
148  pcm.eNumData = OMX_NumericalDataSigned;
149  pcm.eEndian = OMX_EndianLittle;
150  pcm.nSamplingRate = sample_rate;
151  pcm.bInterleaved = OMX_TRUE;
152  pcm.nBitPerSample = bit_depth;
153  pcm.ePCMMode = OMX_AUDIO_PCMModeLinear;
154 
155  switch(num_channels) {
156  case 1:
157  pcm.eChannelMapping[0] = OMX_AUDIO_ChannelCF;
158  break;
159  case 3:
160  pcm.eChannelMapping[2] = OMX_AUDIO_ChannelCF;
161  pcm.eChannelMapping[1] = OMX_AUDIO_ChannelRF;
162  pcm.eChannelMapping[0] = OMX_AUDIO_ChannelLF;
163  break;
164  case 8:
165  pcm.eChannelMapping[7] = OMX_AUDIO_ChannelRS;
166  case 7:
167  pcm.eChannelMapping[6] = OMX_AUDIO_ChannelLS;
168  case 6:
169  pcm.eChannelMapping[5] = OMX_AUDIO_ChannelRR;
170  case 5:
171  pcm.eChannelMapping[4] = OMX_AUDIO_ChannelLR;
172  case 4:
173  pcm.eChannelMapping[3] = OMX_AUDIO_ChannelLFE;
174  pcm.eChannelMapping[2] = OMX_AUDIO_ChannelCF;
175  case 2:
176  pcm.eChannelMapping[1] = OMX_AUDIO_ChannelRF;
177  pcm.eChannelMapping[0] = OMX_AUDIO_ChannelLF;
178  break;
179  }
180 
181  error = OMX_SetParameter(ILC_GET_HANDLE(st->audio_render), OMX_IndexParamAudioPcm, &pcm);
182  assert(error == OMX_ErrorNone);
183 
184  ilclient_change_component_state(st->audio_render, OMX_StateIdle);
185  if(ilclient_enable_port_buffers(st->audio_render, 100, NULL, NULL, NULL) < 0)
186  {
187  // error
188  ilclient_change_component_state(st->audio_render, OMX_StateLoaded);
189  ilclient_cleanup_components(st->list);
190 
191  error = OMX_Deinit();
192  assert(error == OMX_ErrorNone);
193 
194  ilclient_destroy(st->client);
195 
196  sem_destroy(&st->sema);
197  free(st);
198  *handle = NULL;
199  return -1;
200  }
201 
202  ilclient_change_component_state(st->audio_render, OMX_StateExecuting);
203  }
204  }
205 
206  return ret;
207 }

234 {

238 
239  if(hdr)

253 int32_t audioplay_play_buffer(AUDIOPLAY_STATE_T *st,
254  uint8_t *buffer,
255  uint32_t length)
256 {
257  OMX_BUFFERHEADERTYPE *hdr = NULL, *prev = NULL;
258  int32_t ret = -1;
259 
260  if(length % st->bytes_per_sample)
261  return ret;
262 
263  sem_wait(&st->sema);
264 
265  // search through user list for the right buffer header
266  hdr = st->user_buffer_list;
267  while(hdr != NULL && hdr->pBuffer != buffer && hdr->nAllocLen < length)
268  {
269  prev = hdr;
270  hdr = hdr->pAppPrivate;
271  }
272 
273  if(hdr) // we found it, remove from list
274  {
275  ret = 0;
276  if(prev)
277  prev->pAppPrivate = hdr->pAppPrivate;
278  else
279  st->user_buffer_list = hdr->pAppPrivate;
280  }
281 
282  sem_post(&st->sema);
283 
284  if(hdr)
285  {
286  OMX_ERRORTYPE error;
287 
288  hdr->pAppPrivate = NULL;
289  hdr->nOffset = 0;
290  hdr->nFilledLen = length;
291 
292  error = OMX_EmptyThisBuffer(ILC_GET_HANDLE(st->audio_render), hdr);
293  assert(error == OMX_ErrorNone);
294  }
295 
296  return ret;
297 }

341 void play_api_test(int samplerate, int bitdepth, int nchannels, int dest)
342 {
343  AUDIOPLAY_STATE_T *st;
344  int32_t ret;
345  unsigned int i, j, n;
346  int phase = 0;
347  int inc = 256<<16;
348  int dinc = 0;
349  int buffer_size = (BUFFER_SIZE_SAMPLES * bitdepth * OUT_CHANNELS(nchannels))>>3;
350 
351  assert(dest == 0 || dest == 1);
352 
353  ret = audioplay_create(&st, samplerate, nchannels, bitdepth, 10, buffer_size);
354  assert(ret == 0);
355 
356  ret = audioplay_set_dest(st, audio_dest[dest]);
357  assert(ret == 0);
358 
359  // iterate for 5 seconds worth of packets
360  for (n=0; n<((samplerate * 1000)/ BUFFER_SIZE_SAMPLES); n++)
361  {
362  uint8_t *buf;
363  int16_t *p;
364  uint32_t latency;
365 
366  while((buf = audioplay_get_buffer(st)) == NULL)
367  usleep(10*1000);
368 
369  p = (int16_t *) buf;
370 
371  // fill the buffer
372  for (i=0; i<BUFFER_SIZE_SAMPLES; i++)
373  {
374  int16_t val = SIN(phase);
375  phase += inc>>16;
376  inc += dinc;
377  if (inc>>16 < 512)
378  dinc++;
379  else
380  dinc--;
381 
382  for(j=0; j<OUT_CHANNELS(nchannels); j++)
383  {
384  if (bitdepth == 32)
385  *p++ = 0;
386  *p++ = val;
387  }
388  }
389 
390  // try and wait for a minimum latency time (in ms) before
391  // sending the next packet
392  while((latency = audioplay_get_latency(st)) > (samplerate * (MIN_LATENCY_TIME + CTTW_SLEEP_TIME) / 1000))
393  usleep(CTTW_SLEEP_TIME*1000);
394 
395  ret = audioplay_play_buffer(st, buf, buffer_size);
396  assert(ret == 0);
397  }
398 
399  audioplay_delete(st);
400 }
