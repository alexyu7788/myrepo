37 static int32_t gcd(int32_t a, int32_t b)
38 {
39  int shift;
40 
41  if (a == 0 || b == 0)
42  return 1;
43 
44  a = ABS(a);
45  b = ABS(b);
46  for (shift = 0; !((a | b) & 0x01); shift++)
47  a >>= 1, b >>= 1;
48 
49  while (a > 0)
50  {
51  while (!(a & 0x01))
52  a >>= 1;
53  while (!(b & 0x01))
54  b >>= 1;
55  if (a >= b)
56  a = (a - b) >> 1;
57  else
58  b = (b - a) >> 1;
59  }
60  return b << shift;
61 }

64 MMAL_RATIONAL_T mmal_rational_add(MMAL_RATIONAL_T a, MMAL_RATIONAL_T b)
65 {
66  MMAL_RATIONAL_T result;
67  int32_t g = gcd(a.den, b.den);
68  a.den /= g;
69  a.num = a.num * (b.den / g) + b.num * a.den;
70  g = gcd(a.num, g);
71  a.num /= g;
72  a.den *= b.den / g;
73 
74  result.num = a.num;

77 }
