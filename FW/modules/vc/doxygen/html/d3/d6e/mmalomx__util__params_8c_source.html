103 MMAL_STATUS_T mmalomx_param_mapping_generic(MMALOMX_PARAM_MAPPING_DIRECTION dir,
104  const MMALOMX_PARAM_TRANSLATION_T *xlat,
105  MMAL_PARAMETER_HEADER_T *mmal_param, OMX_PTR omx_param, MMAL_PORT_T *mmal_port)
106 {
107  MMALOMX_PARAM_OMX_HEADER_T *omx_header = (MMALOMX_PARAM_OMX_HEADER_T *)omx_param;
108  uint8_t *mmal_data = ((uint8_t *)mmal_param) + sizeof(MMAL_PARAMETER_HEADER_T);
109  uint8_t *omx_data = ((uint8_t *)omx_param) + sizeof(MMALOMX_PARAM_OMX_HEADER_T);
110  unsigned int size = mmal_param->size - sizeof(MMAL_PARAMETER_HEADER_T);
111  MMAL_PARAM_UNUSED(mmal_port);
112 
113  if (xlat->portless)
114  omx_data -= sizeof(OMX_U32);
115 
116  if (((uint8_t *)omx_param) + omx_header->nSize !=
117  omx_data + size)
118  {
119  VCOS_ALERT("mmalomx_param_mapping_generic: mismatch between mmal and omx parameters for (%u)",
120  (unsigned int)mmal_param->id);
121  return MMAL_EINVAL;
122  }
123 
124  if (dir == MMALOMX_PARAM_MAPPING_TO_MMAL)
125  memcpy(mmal_data, omx_data, size);
126  else
127  memcpy(omx_data, mmal_data, size);
128 
129  return MMAL_SUCCESS;
130 }

132 MMAL_STATUS_T mmalomx_param_enum_generic(MMALOMX_PARAM_MAPPING_DIRECTION dir,
133  const MMALOMX_PARAM_TRANSLATION_T *xlat,
134  MMAL_PARAMETER_HEADER_T *mmal_param, OMX_PTR omx_param, MMAL_PORT_T *mmal_port)
135 {
136  uint32_t *mmal = (uint32_t *)(((uint8_t *)mmal_param) + sizeof(MMAL_PARAMETER_HEADER_T));
137  uint32_t *omx = (uint32_t *)(((uint8_t *)omx_param) + sizeof(MMALOMX_PARAM_OMX_HEADER_T));
138  unsigned int i = 0;
139  MMAL_PARAM_UNUSED(mmal_port);
140 
141  if (xlat->portless)
142  omx -= 1;
143 
144  /* Find translation entry in lookup table */
145  if (dir == MMALOMX_PARAM_MAPPING_TO_MMAL)
146  for (i = 0; i < xlat->xlat_enum_num && xlat->xlat_enum->omx != *omx; i++);
147  else
148  for (i = 0; i < xlat->xlat_enum_num && xlat->xlat_enum->mmal != *mmal; i++);
149 
150  if (i == xlat->xlat_enum_num)
151  {
152  if (dir == MMALOMX_PARAM_MAPPING_TO_MMAL)
153  VCOS_ALERT("mmalomx_param_enum_generic: omx enum value %u not supported", (unsigned int)*omx);
154  else
155  VCOS_ALERT("mmalomx_param_enum_generic: mmal enum value %u not supported", (unsigned int)*mmal);
156  return MMAL_EINVAL;
157  }
158 
159  if (dir == MMALOMX_PARAM_MAPPING_TO_MMAL)
160  *mmal = xlat->xlat_enum[i].mmal;
161  else
162  *omx = xlat->xlat_enum[i].omx;
163 
164  return MMAL_SUCCESS;
165 }
