65 static void sdl_callback( void *ctx, uint8_t *stream, int size )
66 {
67  MMAL_PORT_T *port = (MMAL_PORT_T *)ctx;
68  MMAL_COMPONENT_T *component = port->component;
69  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
70  MMAL_BUFFER_HEADER_T *buffer;
71  unsigned int i, bytes;
72 
73  while (size > 0)
74  {
75  buffer = mmal_queue_get(module->queue);
76  if (!buffer)
77  {
78  LOG_ERROR("audio underrun");
79  return;
80  }
81 
82  if (port->format->encoding == MMAL_ENCODING_PCM_SIGNED &&
83  port->format->es->audio.bits_per_sample == 16)
84  {
85  bytes = buffer->length;
86  if (bytes > (unsigned int)size) bytes = size;
87  memcpy(stream, buffer->data + buffer->offset, bytes);
88  buffer->offset += bytes;
89  buffer->length -= bytes;
90  stream += bytes;
91  size -= bytes;
92  }
93  else if (port->format->es->audio.bits_per_sample == 32)
94  {
95  bytes = buffer->length;
96  if (bytes > 2 * (unsigned int)size) bytes = 2 * size;
97  vcos_assert(!(bytes&0x3));
98 
99  if (port->format->encoding == MMAL_ENCODING_PCM_FLOAT)
100  {
101  float *in = (float *)(buffer->data + buffer->offset);
102  int16_t *out = (int16_t *)stream;
103  for (i = 0; i < bytes / 4; i++)
104  {
105  if (*in >= 1.0) *out = 32767;
106  else if (*in < -1.0) *out = -32768;
107  else *out = *in * 32768.0;
108  in++; out++;
109  }
110  }
111  else if (port->format->encoding == MMAL_ENCODING_PCM_SIGNED)
112  {
113  int32_t *in = (int32_t *)(buffer->data + buffer->offset);
114  int16_t *out = (int16_t *)stream;
115  for (i = 0; i < bytes / 4; i++)
116  *out++ = (*in++) >> 16;
117  }
118  buffer->offset += bytes;
119  buffer->length -= bytes;
120  stream += bytes / 2;
121  size -= bytes / 2;
122  }
123 
124  if (buffer->length)
125  {
126  /* We still have some data left for next time */
127  mmal_queue_put_back(module->queue, buffer);
128  continue;
129  }
130 
131  /* Handle the EOS */
132  if (buffer->flags & MMAL_BUFFER_HEADER_FLAG_EOS)
133  mmal_event_eos_send(port);
134 
135  buffer->offset = 0;
136  mmal_port_buffer_header_callback(port, buffer);
137  }
138 }

141 static MMAL_STATUS_T sdl_port_set_format(MMAL_PORT_T *port)
142 {
143  MMAL_COMPONENT_T *component = port->component;
144  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
145  SDL_AudioSpec desired, obtained;
146 
147  if (port->format->encoding != MMAL_ENCODING_PCM_SIGNED &&
148  port->format->encoding != MMAL_ENCODING_PCM_FLOAT &&
149  port->format->es->audio.bits_per_sample != 16 &&
150  port->format->es->audio.bits_per_sample != 32)
151  {
152  LOG_ERROR("port does not support '%4.4s' at %ibps",
153  (char *)&port->format->encoding, port->format->es->audio.bits_per_sample);
154  return MMAL_EINVAL;
155  }
156 
157  if (module->audio_opened)
158  SDL_CloseAudio();
159  module->audio_opened = MMAL_FALSE;
160 
161  desired.freq = port->format->es->audio.sample_rate;
162  desired.format = AUDIO_S16SYS;
163  desired.channels = port->format->es->audio.channels;
164  desired.callback = sdl_callback;
165  desired.userdata = port;
166  desired.samples = FRAME_LENGTH;
167 
168  /* Open the sound device. */
169  if (SDL_OpenAudio( &desired, &obtained ) < 0)
170  return MMAL_ENOSYS;
171  module->audio_opened = MMAL_TRUE;
172 
173  /* Now have a look at what we got. */
174  if (obtained.format != AUDIO_S16SYS)
175  return MMAL_ENOSYS;
176 
177  port->format->es->audio.sample_rate = obtained.freq;
178  port->format->es->audio.channels = obtained.channels;
179  port->buffer_size_min = obtained.samples * port->format->es->audio.channels * 2;
180 
181  return MMAL_SUCCESS;
182 }

207 static MMAL_STATUS_T sdl_port_send(MMAL_PORT_T *port, MMAL_BUFFER_HEADER_T *buffer)
208 {
209  MMAL_COMPONENT_T *component = port->component;
210  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
211 
212  /* Handle event buffers */
213  if (buffer->cmd)
214  {
215  LOG_ERROR("discarding event %i on port %p", (int)buffer->cmd, port);
216  buffer->length = 0;
217  mmal_port_buffer_header_callback(port, buffer);
218  return MMAL_SUCCESS;
219  }
220 
221  if (module->status != MMAL_SUCCESS)
222  return module->status;
223 
224  mmal_queue_put(module->queue, buffer);
225 
226  return MMAL_SUCCESS;
227 }

230 static MMAL_STATUS_T mmal_component_create_sdl(const char *name, MMAL_COMPONENT_T *component)
231 {
232  MMAL_COMPONENT_MODULE_T *module;
233 
234  /* Check we're the requested component */
235  if(strcmp(name, "sdl." MMAL_AUDIO_RENDER))
236  return MMAL_ENOENT;
237 
238  if( SDL_WasInit(SDL_INIT_AUDIO) )
239  return MMAL_ENXIO;
240 
241  /* Allocate our module context */
242  component->priv->module = module = vcos_calloc(1, sizeof(*module), "mmal module");
243  if(!module)
244  return MMAL_ENOMEM;
245 
246  if(SDL_Init(SDL_INIT_AUDIO|SDL_INIT_NOPARACHUTE) < 0)
247  return MMAL_ENXIO;
248 
249  /* Allocate the ports for this component */
250  component->input = mmal_ports_alloc(component, 1, MMAL_PORT_TYPE_INPUT, 0);
251  if(!component->input)
252  goto error;
253  component->input_num = 1;
254 
255  module->queue = mmal_queue_create();
256  if(!module->queue)
257  goto error;
258 
259  component->input[0]->priv->pf_set_format = sdl_port_set_format;
260  component->input[0]->priv->pf_enable = sdl_port_enable;
261  component->input[0]->priv->pf_disable = sdl_port_disable;
262  component->input[0]->priv->pf_send = sdl_port_send;
263  component->input[0]->buffer_num_min = INPUT_MIN_BUFFER_NUM;
264  component->input[0]->buffer_num_recommended = INPUT_RECOMMENDED_BUFFER_NUM;
265 
266  component->priv->pf_destroy = sdl_component_destroy;
267  return MMAL_SUCCESS;
268 
269  error:
270  sdl_component_destroy(component);
271  return MMAL_ENOMEM;
272 }
