411 /* Request notification when a reference to the image can be acquired.
412  *
413  * If the reference can already be acquired at the time when the request is made,
414  * this is indicated via the return code, and no callback will be made.
415  *
416  * Otherwise the callback is made when the reference can be acquired, or when
417  * the specified time period has elapsed.
418  *
419  * The thread context in which callbacks are executed depends upon the converter
420  * class implementation. The callback function should therefore do as little
421  * work as possible, deferring to a known thread context for any further
422  * processing.
423  *
424  * @param converter converter functions for this image
425  * @param image image pointer
426  * @param timeout time to wait in milliseconds; 0 means wait indefinitely
427  * @param callback callback to be invoked when image is valid, or timeout expires
428  *
429  * @return IMAGECONV_ERR_NOT_SUPPORTED if not implemented by the converter class
430  * @return IMAGECONV_ERR_ALREADY if reference can be acquired immediately
431  * @return IMAGECONV_ERR_NONE if reference cannot be acquire immediately; callback will be made
432  */
433 int imageconv_notify(const IMAGE_CONVERT_CLASS_T *converter,
434  IMAGECONV_DRIVER_IMAGE_T *image,
435  uint32_t timeout,
436  IMAGECONV_NOTIFY_CALLBACK callback,
437  void *context);
