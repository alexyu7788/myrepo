50 /**********************************************************************\
51 |* Demonstration program to compute the 32-bit CRC used as the frame *|
52 |* check sequence in ADCCP (ANSI X3.66, also known as FIPS PUB 71 *|
53 |* and FED-STD-1003, the U.S. versions of CCITT's X.25 link-level *|
54 |* protocol). The 32-bit FCS was added via the Federal Register, *|
55 |* 1 June 1982, p.23798. I presume but don't know for certain that *|
56 |* this polynomial is or will be included in CCITT V.41, which *|
57 |* defines the 16-bit CRC (often called CRC-CCITT) polynomial. FIPS *|
58 |* PUB 78 says that the 32-bit FCS reduces otherwise undetected *|
59 |* errors by a factor of 10^-5 over 16-bit FCS. *|
60 \**********************************************************************/

69 /* First, the polynomial itself and its table of feedback terms. The */
70 /* polynomial is */
71 /* X^32+X^26+X^23+X^22+X^16+X^12+X^11+X^10+X^8+X^7+X^5+X^4+X^2+X^1+X^0 */
72 /* Note that we take it "backwards" and put the highest-order term in */
73 /* the lowest-order bit. The X^32 term is "implied"; the LSB is the */
74 /* X^31 term, etc. The X^0 term (usually shown as "+1") results in */
75 /* the MSB being 1. */
76 
77 /* Note that the usual hardware shift register implementation, which */
78 /* is what we're using (we're merely optimizing it by doing eight-bit */
79 /* chunks at a time) shifts bits into the lowest-order term. In our */
80 /* implementation, that means shifting towards the right. Why do we */
81 /* do it this way? Because the calculated CRC must be transmitted in */
82 /* order from highest-order term to lowest-order term. UARTs transmit */
83 /* characters in order from LSB to MSB. By storing the CRC this way, */
84 /* we hand it to the UART in the order low-byte to high-byte; the UART */
85 /* sends each low-bit to hight-bit; and the result is transmission bit */
86 /* by bit from highest- to lowest-order term without requiring any bit */
87 /* shuffling on our part. Reception works similarly. */
88 
89 /* The feedback terms table consists of 256, 32-bit entries. Notes: */
90 /* */
91 /* 1. The table can be generated at runtime if desired; code to do so */
92 /* is shown later. It might not be obvious, but the feedback */
93 /* terms simply represent the results of eight shift/xor opera- */
94 /* tions for all combinations of data and CRC register values. */
95 /* */
96 /* 2. The CRC accumulation logic is the same for all CRC polynomials, */
97 /* be they sixteen or thirty-two bits wide. You simply choose the */
98 /* appropriate table. Alternatively, because the table can be */
99 /* generated at runtime, you can start by generating the table for */
100 /* the polynomial in question and use exactly the same "updcrc", */
101 /* if your application needn't simultaneously handle two CRC */
102 /* polynomials. (Note, however, that XMODEM is strange.) */
103 /* */
104 /* 3. For 16-bit CRCs, the table entries need be only 16 bits wide; */
105 /* of course, 32-bit entries work OK if the high 16 bits are zero. */
106 /* */
107 /* 4. The values must be right-shifted by eight bits by the "updcrc" */
108 /* logic; the shift must be unsigned (bring in zeroes). On some */
109 /* hardware you could probably optimize the shift in assembler by */
110 /* using byte-swap instructions. */

163 Boolean_T crc32file(char *name, uint32_t *crc, long *charcnt)
164 {
165  register FILE *fin;
166  register uint32_t oldcrc32;
167  register int c;
168 
169  oldcrc32 = 0xFFFFFFFF; *charcnt = 0;
170 #ifdef MSDOS
171  if ((fin=fopen(name, "rb"))==NULL)
172 #else
173  if ((fin=fopen(name, "r"))==NULL)
174 #endif
175  {
176  perror(name);
177  return Error_;
178  }
179  while ((c=getc(fin))!=EOF)
180  {
181  ++*charcnt;
182  oldcrc32 = UPDC32(c, oldcrc32);
183  }
184 
185  if (ferror(fin))
186  {
187  perror(name);
188  *charcnt = -1;
189  }
190  fclose(fin);
191 
192  *crc = oldcrc32 = ~oldcrc32;
193 
194  return Success_;
195 }
