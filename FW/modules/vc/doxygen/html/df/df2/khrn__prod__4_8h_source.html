197 static INLINE void put_byte(uint8_t *p, uint8_t n)
198 {
199  #ifdef BIG_ENDIAN_CPU

201  #else

204 }
205 static INLINE void put_short(uint8_t *p, uint16_t n)
206 {
207  #ifdef BIG_ENDIAN_CPU
208  *(uint8_t*)((uint32_t)&p[0]^0x3) = (uint8_t)n;
209  *(uint8_t*)((uint32_t)&p[1]^0x3) = (uint8_t)(n >> 8);
210  #else
211  p[0] = (uint8_t)n;
212  p[1] = (uint8_t)(n >> 8);
213  #endif
214 }

216 static INLINE void put_word(uint8_t *p, uint32_t n)
217 {
218  #ifdef BIG_ENDIAN_CPU

220  *(uint8_t*)((uint32_t)&p[1]^0x3) = (uint8_t)(n >> 8);
221  *(uint8_t*)((uint32_t)&p[2]^0x3) = (uint8_t)(n >> 16);
222  *(uint8_t*)((uint32_t)&p[3]^0x3) = (uint8_t)(n >> 24);
223  #else
224  p[0] = (uint8_t)n;
225  p[1] = (uint8_t)(n >> 8);
226  p[2] = (uint8_t)(n >> 16);
227  p[3] = (uint8_t)(n >> 24);
228  #endif

376 /*
377  if we don't run out of memory at any point, callback will be called:
378  - after locking everything in place, with reason
379  KHRN_HW_CALLBACK_REASON_FIXUP, from the llat task. khrn_hw_fixup_done
380  should be called at some point after this to signal that fixup is complete
381  and binning can begin
382  - after binning has completed, first with reason
383  KHRN_HW_CALLBACK_REASON_BIN_FINISHED_LLAT from the llat task, then with
384  KHRN_HW_CALLBACK_REASON_BIN_FINISHED from the master task
385  - after rendering has completed, with reason
386  KHRN_HW_CALLBACK_REASON_RENDER_FINISHED, from the master task
387 
388  we can run out of memory before or during binning (but not after). if we do
389  run out of memory, callback will be called first with reason
390  KHRN_HW_CALLBACK_REASON_OUT_OF_MEM_LLAT from the llat task, then with reason
391  KHRN_HW_CALLBACK_REASON_OUT_OF_MEM from the master task, after which it will
392  not be called again
393 */
394

399  KHRN_HW_CALLBACK_REASON_BIN_FINISHED_LLAT,
400  KHRN_HW_CALLBACK_REASON_BIN_FINISHED,
401  KHRN_HW_CALLBACK_REASON_RENDER_FINISHED,
