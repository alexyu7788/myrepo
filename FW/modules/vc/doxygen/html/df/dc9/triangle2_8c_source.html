203 static void init_shaders(CUBE_STATE_T *state)
204 {
205  static const GLfloat vertex_data[] = {
206  -1.0,-1.0,1.0,1.0,
207  1.0,-1.0,1.0,1.0,
208  1.0,1.0,1.0,1.0,
209  -1.0,1.0,1.0,1.0
210  };
211  const GLchar *vshader_source =
212  "attribute vec4 vertex;"
213  "varying vec2 tcoord;"
214  "void main(void) {"
215  " vec4 pos = vertex;"
216  " gl_Position = pos;"
217  " tcoord = vertex.xy*0.5+0.5;"
218  "}";
219 
220  //Mandelbrot
221  const GLchar *mandelbrot_fshader_source =
222 "uniform vec4 color;"
223 "uniform vec2 scale;"
224 "uniform vec2 centre;"
225 "varying vec2 tcoord;"
226 "void main(void) {"
227 " float intensity;"
228 " vec4 color2;"
229 " float cr=(gl_FragCoord.x-centre.x)*scale.x;"
230 " float ci=(gl_FragCoord.y-centre.y)*scale.y;"
231 " float ar=cr;"
232 " float ai=ci;"
233 " float tr,ti;"
234 " float col=0.0;"
235 " float p=0.0;"
236 " int i=0;"
237 " for(int i2=1;i2<16;i2++)"
238 " {"
239 " tr=ar*ar-ai*ai+cr;"
240 " ti=2.0*ar*ai+ci;"
241 " p=tr*tr+ti*ti;"
242 " ar=tr;"
243 " ai=ti;"
244 " if (p>16.0)"
245 " {"
246 " i=i2;"
247 " break;"
248 " }"
249 " }"
250 " color2 = vec4(float(i)*0.0625,0,0,1);"
251 " gl_FragColor = color2;"
252 "}";
253 
254  // Julia
255  const GLchar *julia_fshader_source =
256 "uniform vec4 color;"
257 "uniform vec2 scale;"
258 "uniform vec2 centre;"
259 "uniform vec2 offset;"
260 "varying vec2 tcoord;"
261 "uniform sampler2D tex;"
262 "void main(void) {"
263 " float intensity;"
264 " vec4 color2;"
265 " float ar=(gl_FragCoord.x-centre.x)*scale.x;"
266 " float ai=(gl_FragCoord.y-centre.y)*scale.y;"
267 " float cr=(offset.x-centre.x)*scale.x;"
268 " float ci=(offset.y-centre.y)*scale.y;"
269 " float tr,ti;"
270 " float col=0.0;"
271 " float p=0.0;"
272 " int i=0;"
273 " vec2 t2;"
274 " t2.x=tcoord.x+(offset.x-centre.x)*(0.5/centre.y);"
275 " t2.y=tcoord.y+(offset.y-centre.y)*(0.5/centre.x);"
276 " for(int i2=1;i2<16;i2++)"
277 " {"
278 " tr=ar*ar-ai*ai+cr;"
279 " ti=2.0*ar*ai+ci;"
280 " p=tr*tr+ti*ti;"
281 " ar=tr;"
282 " ai=ti;"
283 " if (p>16.0)"
284 " {"
285 " i=i2;"
286 " break;"
287 " }"
288 " }"
289 " color2 = vec4(0,float(i)*0.0625,0,1);"
290 " color2 = color2+texture2D(tex,t2);"
291 " gl_FragColor = color2;"
292 "}";

364  check();

374  check();

376  check();

378  glViewport ( 0, 0, state->screen_width, state->screen_height );
379  check();
380

382  glBindBuffer(GL_ARRAY_BUFFER, state->buf);
383  glBufferData(GL_ARRAY_BUFFER, sizeof(vertex_data),
384  vertex_data, GL_STATIC_DRAW);

386  glEnableVertexAttribArray(state->attr_vertex);
387  glVertexAttribPointer(state->attr_vertex2, 4, GL_FLOAT, 0, 16, 0);
388  glEnableVertexAttribArray(state->attr_vertex2);
389 
390  check();
391 }

449 static int get_mouse(CUBE_STATE_T *state, int *outx, int *outy)
450 {
451  static int fd = -1;
452  const int width=state->screen_width, height=state->screen_height;
453  static int x=800, y=400;
454  const int XSIGN = 1<<4, YSIGN = 1<<5;
455  if (fd<0) {
456  fd = open("/dev/input/mouse0",O_RDONLY|O_NONBLOCK);
457  }
458  if (fd>=0) {
459  struct {char buttons, dx, dy; } m;
460  while (1) {
461  int bytes = read(fd, &m, sizeof m);
462  if (bytes < (int)sizeof m) goto _exit;
463  if (m.buttons&8) {
464  break; // This bit should always be set
465  }
466  read(fd, &m, 1); // Try to sync up again
467  }
468  if (m.buttons&3)
469  return m.buttons&3;
470  x+=m.dx;
471  y+=m.dy;
472  if (m.buttons&XSIGN)
473  x-=256;
474  if (m.buttons&YSIGN)
475  y-=256;
476  if (x<0) x=0;
477  if (y<0) y=0;
478  if (x>width) x=width;
479  if (y>height) y=height;
480  }
481 _exit:
482  if (outx) *outx = x;
483  if (outy) *outy = y;
484  return 0;
485 }
