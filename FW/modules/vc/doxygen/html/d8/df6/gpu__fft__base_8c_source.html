57 int gpu_fft_get_host_info(struct GPU_FFT_HOST *info) {
58  void *handle;
59  unsigned (*bcm_host_get_sdram_address) (void);
60  unsigned (*bcm_host_get_peripheral_address)(void);
61  unsigned (*bcm_host_get_peripheral_size) (void);
62 
63  // Pi 1 defaults
64  info->peri_addr = 0x20000000;
65  info->peri_size = 0x01000000;
66  info->mem_flg = GPU_FFT_USE_VC4_L2_CACHE? 0xC : 0x4;
67  info->mem_map = GPU_FFT_USE_VC4_L2_CACHE? 0x0 : 0x20000000; // Pi 1 only
68 
69  handle = dlopen("libbcm_host.so", RTLD_LAZY);
70  if (!handle) return -1;
71 
72  *(void **) (&bcm_host_get_sdram_address) = dlsym(handle, "bcm_host_get_sdram_address");
73  *(void **) (&bcm_host_get_peripheral_address) = dlsym(handle, "bcm_host_get_peripheral_address");
74  *(void **) (&bcm_host_get_peripheral_size) = dlsym(handle, "bcm_host_get_peripheral_size");
75 
76  if (bcm_host_get_sdram_address && bcm_host_get_sdram_address()!=0x40000000) { // Pi 2?
77  info->mem_flg = 0x4; // ARM cannot see VC4 L2 on Pi 2
78  info->mem_map = 0x0;
79  }
80 
81  if (bcm_host_get_peripheral_address) info->peri_addr = bcm_host_get_peripheral_address();
82  if (bcm_host_get_peripheral_size) info->peri_size = bcm_host_get_peripheral_size();
83 
84  dlclose(handle);
85  return 0;
86 }

88 unsigned gpu_fft_base_exec_direct (
89  struct GPU_FFT_BASE *base,
90  int num_qpus) {
91 
92  unsigned q, t;
93 
94  base->peri[V3D_DBCFG] = 0; // Disallow IRQ
95  base->peri[V3D_DBQITE] = 0; // Disable IRQ
96  base->peri[V3D_DBQITC] = -1; // Resets IRQ flags
97 
98  base->peri[V3D_L2CACTL] = 1<<2; // Clear L2 cache
99  base->peri[V3D_SLCACTL] = -1; // Clear other caches
100 
101  base->peri[V3D_SRQCS] = (1<<7) | (1<<8) | (1<<16); // Reset error bit and counts
102 
103  for (q=0; q<num_qpus; q++) { // Launch shader(s)
104  base->peri[V3D_SRQUA] = base->vc_unifs[q];
105  base->peri[V3D_SRQPC] = base->vc_code;
106  }
107 
108  // Busy wait polling
109  for (;;) {
110  if (((base->peri[V3D_SRQCS]>>16) & 0xff) == num_qpus) break; // All done?
111  }
112 
113  return 0;
114 }

117  struct GPU_FFT_BASE *base,

120  if (base->vc_msg) {

131 int gpu_fft_alloc (
132  int mb,
133  unsigned size,
134  struct GPU_FFT_PTR *ptr) {
135 
136  struct GPU_FFT_HOST host;
137  struct GPU_FFT_BASE *base;
138  volatile unsigned *peri;
139  unsigned handle;
140 
141  if (gpu_fft_get_host_info(&host)) return -5;
142 
143  if (qpu_enable(mb, 1)) return -1;
144 
145  // Shared memory
146  handle = mem_alloc(mb, size, 4096, host.mem_flg);
147  if (!handle) {
148  qpu_enable(mb, 0);
149  return -3;
150  }
151 
152  peri = (volatile unsigned *) mapmem(host.peri_addr, host.peri_size);
153  if (!peri) {
154  mem_free(mb, handle);
155  qpu_enable(mb, 0);
156  return -4;
157  }
158 
159  ptr->vc = mem_lock(mb, handle);
160  ptr->arm.vptr = mapmem(BUS_TO_PHYS(ptr->vc+host.mem_map), size);
161 
162  base = (struct GPU_FFT_BASE *) ptr->arm.vptr;
163  base->peri = peri;
164  base->peri_size = host.peri_size;
165  base->mb = mb;
166  base->handle = handle;
167  base->size = size;
168 
169  return 0;
170 }
