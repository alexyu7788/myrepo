125 /*
126  A MEM_HANDLE_T refers to a movable block of memory.
127 
128  The only way to get a MEM_HANDLE_T is to call mem_alloc.
129 
130  The MEM_HANDLE_T you get is immutable and remains valid until its reference
131  count reaches 0.
132 
133  A MEM_HANDLE_T has an initial reference count of 1. This can be incremented
134  by calling mem_acquire and decremented by calling mem_release.
135 */
136 
137 /*
138  MEM_ZERO_SIZE_HANDLE is a preallocated handle to a zero-size block of memory
139  MEM_EMPTY_STRING_HANDLE is a preallocated handle to a block of memory containing the empty string
140 
141  MEM_HANDLE_INVALID is the equivalent of NULL for MEM_HANDLE_Ts -- no valid
142  MEM_HANDLE_T will ever equal MEM_HANDLE_INVALID.
143 */
144

151  MEM_EMPTY_STRING_HANDLE = 2,
152 #endif

154  MEM_HANDLE_INVALID = 0,

168 typedef enum {

171  /*
172  If a MEM_HANDLE_T is discardable, the memory manager may resize it to size

176  MEM_FLAG_DISCARDABLE = 1 << 0,

178  /*
179  MEM_FLAG_RETAINED should only ever be used when passing flags to

183  MEM_FLAG_RETAINED = 1 << 9, /* shared with MEM_FLAG_ABANDONED. only used when passing flags to mem_alloc */

185  /*
186  Block must be kept within bottom 256M region of the relocatable heap.
187  Specifying this flag means that an allocation will fail if the block
188  cannot be allocated within that region, and the block will not be moved
189  out of that range.
190  (This is to support memory blocks used by the codec cache, which must
191  have same top 4 bits; see HW-3058)
192  This flag is ignored on non-VideoCore platforms.
193  */
194 
195  MEM_FLAG_LOW_256M = 1 << 1,
196 
197  /*
198  Define a mask to extract the memory alias used by the block of memory.
199  */
200 
201  MEM_FLAG_ALIAS_MASK = 3 << 2,
202 
203  /*
204  If a MEM_HANDLE_T is allocating (or normal), its block of memory will be
205  accessed in an allocating fashion through the cache.
206  */
207 
208  MEM_FLAG_NORMAL = 0 << 2,
209  MEM_FLAG_ALLOCATING = MEM_FLAG_NORMAL,
210 
211  /*
212  If a MEM_HANDLE_T is direct, its block of memory will be accessed
213  directly, bypassing the cache.
214  */
215 
216  MEM_FLAG_DIRECT = 1 << 2,
217 
218  /*
219  If a MEM_HANDLE_T is coherent, its block of memory will be accessed in a
220  non-allocating fashion through the cache.
221  */
222 
223  MEM_FLAG_COHERENT = 2 << 2,
224 
225  /*
226  If a MEM_HANDLE_T is L1-nonallocating, its block of memory will be accessed by
227  the VPU in a fashion which is allocating in L2, but only coherent in L1.
228  */
229 
230  MEM_FLAG_L1_NONALLOCATING = (MEM_FLAG_DIRECT | MEM_FLAG_COHERENT),
231 
232  /*
233  If a MEM_HANDLE_T is zero'd, its contents are set to 0 rather than
234  MEM_HANDLE_INVALID on allocation and resize up.
235  */
236 
237  MEM_FLAG_ZERO = 1 << 4,
238 
239  /*
240  If a MEM_HANDLE_T is uninitialised, it will not be reset to a defined value
241  (either zero, or all 1's) on allocation.
242  */
243 
244  MEM_FLAG_NO_INIT = 1 << 5,
245 
246  /*
247  The INIT flag is a placeholder, designed to make it explicit that
248  initialisation is required, and to make it possible to change the sense
249  of this bit at a later date.
250  */
251 
252  MEM_FLAG_INIT = 0 << 5,
253 
254  /*
255  Hints.
256  */
257 
258  MEM_FLAG_HINT_PERMALOCK = 1 << 6, /* Likely to be locked for long periods of time. */
259 
260  MEM_FLAG_HINT_ALL = 0xc0,
261 
262  MEM_FLAG_USER = 1 << 7,
263  MEM_FLAG_HINT_GROW = 1 << 7, /* Likely to grow in size over time. If this flag is specified, MEM_FLAG_RESIZEABLE must also be. */
264 
265  MEM_FLAG_UNUSED = 1 << 7,
266 
267  /*
268  If a MEM_HANDLE_T is to be resized with mem_resize, this flag must be
269  present. This flag prevents things from being allocated out of the small
270  allocation pool.
271  */
272  MEM_FLAG_RESIZEABLE = 1 << 8,
273 
274  /*
275  If the ABANDONED flag is set, because mem_abandon was called when the lock
276  count was zero, the contents are undefined. This flag is cleared by
277  mem_lock. Automatically set for blocks allocated with MEM_FLAG_NO_INIT.

279 
280  MEM_FLAG_ABANDONED = 1 << 9, /* shared with MEM_FLAG_RETAINED. never used when passing flags to mem_alloc */
