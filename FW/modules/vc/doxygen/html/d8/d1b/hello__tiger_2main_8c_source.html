91 PS* PS_construct(const char* commands, int commandCount, const float* points, int pointCount)
92 {

94  int p = 0;
95  int c = 0;
96  int i = 0;

98  int maxElements = 0;
99  unsigned char* cmd;
100  UNREF(pointCount);
101 
102  while(c < commandCount)
103  {
104  int elements, e;
105  c += 4;
106  p += 8;
107  elements = (int)points[p++];
108  assert(elements > 0);
109  if(elements > maxElements)
110  maxElements = elements;
111  for(e=0;e<elements;e++)
112  {
113  switch(commands[c])
114  {
115  case 'M': p += 2; break;
116  case 'L': p += 2; break;
117  case 'C': p += 6; break;
118  case 'E': break;
119  default:
120  assert(0); //unknown command
121  }
122  c++;
123  }
124  paths++;
125  }
126 
127  ps->m_numPaths = paths;
128  ps->m_paths = (PathData*)malloc(paths * sizeof(PathData));
129  cmd = (unsigned char*)malloc(maxElements);
130 
131  i = 0;
132  p = 0;
133  c = 0;
134  while(c < commandCount)
135  {
136  int elements, startp, e;
137  float color[4];
138 
139  //fill type
140  int paintMode = 0;
141  ps->m_paths[i].m_fillRule = VG_NON_ZERO;
142  switch( commands[c] )
143  {
144  case 'N':
145  break;
146  case 'F':
147  ps->m_paths[i].m_fillRule = VG_NON_ZERO;
148  paintMode |= VG_FILL_PATH;
149  break;

209  ps->m_paths[i].m_strokeWidth = points[p++];
210 
211  //paints

213  color[1] = points[p++];

227 
228  //read number of elements
229 
230  elements = (int)points[p++];

232  startp = p;
233  for(e=0;e<elements;e++)
234  {
235  switch( commands[c] )
236  {
237  case 'M':

260  i++;
261  }

298  }

309 void render(int w, int h)
310 {
311 #ifndef __RASPBERRYPI__
312  if(renderWidth != w || renderHeight != h)
313 #endif
314  {
315  float clearColor[4] = {1,1,1,1};
316  float scale = w / (tigerMaxX - tigerMinX);
317 
318  eglSwapBuffers(egldisplay, eglsurface); //force EGL to recognize resize
319 
320  vgSetfv(VG_CLEAR_COLOR, 4, clearColor);
321  vgClear(0, 0, w, h);
322 
323  vgLoadIdentity();
324 #ifdef __RASPBERRYPI__
325  vgTranslate(w * 0.5f, h * 0.5f);
326  vgRotate(rotateN);
327  vgTranslate(-w * 0.5f, -h * 0.5f);
328 #endif
329  vgScale(scale, scale);
330  vgTranslate(-tigerMinX, -tigerMinY + 0.5f * (h / scale - (tigerMaxY - tigerMinY)));
331 
332  PS_render(tiger);
333  assert(vgGetError() == VG_NO_ERROR);
334 
335  renderWidth = w;
336  renderHeight = h;
337  }
338 #ifndef __RASPBERRYPI__
339  eglSwapBuffers(egldisplay, eglsurface);
340  assert(eglGetError() == EGL_SUCCESS);
341 #endif
342 }
