52 int32_t render_subtitle(GRAPHICS_RESOURCE_HANDLE img, const char *text, const int skip, const uint32_t text_size, const uint32_t y_offset)
53 {
54  uint32_t text_length = strlen(text)-skip;
55  uint32_t width=0, height=0;
56  const char *split = text;
57  int32_t s=0;
58  int len = 0; // length of pre-subtitle
59  uint32_t img_w, img_h;
60 
61  graphics_get_resource_size(img, &img_w, &img_h);
62 
63  if (text_length==0)
64  return 0;
65  while (split[0]) {
66  s = graphics_resource_text_dimensions_ext(img, split, text_length-(split-text), &width, &height, text_size);
67  if (s != 0) return s;
68  if (width > img_w) {
69  const char *space = strnchr(split, text_length-(split-text), ' ');
70  if (!space) {
71  len = split+1-text;
72  split = split+1;
73  } else {
74  len = space-text;
75  split = space+1;
76  }
77  } else {
78  break;
79  }
80  }
81  // split now points to last line of text. split-text = length of initial text. text_length-(split-text) is length of last line
82  if (width) {
83  s = graphics_resource_render_text_ext(img, (img_w - width)>>1, y_offset-height,
84  GRAPHICS_RESOURCE_WIDTH,
85  GRAPHICS_RESOURCE_HEIGHT,
86  GRAPHICS_RGBA32(0xff,0xff,0xff,0xff), /* fg */
87  GRAPHICS_RGBA32(0,0,0,0x80), /* bg */
88  split, text_length-(split-text), text_size);
89  if (s!=0) return s;
90  }
91  return render_subtitle(img, text, skip+text_length-len, text_size, y_offset - height);
92 }
