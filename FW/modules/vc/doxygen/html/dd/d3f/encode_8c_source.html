43 // generate an animated test card in YUV format
44 static int
45 generate_test_card(void *buf, OMX_U32 * filledLen, int frame, OMX_PARAM_PORTDEFINITIONTYPE *def)
46 {
47  int i, j;
48  OMX_VIDEO_PORTDEFINITIONTYPE *vid = &def->format.video;
49  char *y = buf;
50  char *u = y + vid->nStride * vid->nSliceHeight;
51  char *v = u + (vid->nStride >> 1) * (vid->nSliceHeight >> 1);
52 
53  for (j = 0; j < vid->nFrameHeight / 2; j++) {
54  char *py = y + 2 * j * vid->nStride;
55  char *pu = u + j * (vid->nStride >> 1);
56  char *pv = v + j * (vid->nStride >> 1);
57  for (i = 0; i < vid->nFrameWidth / 2; i++) {
58  int z = (((i + frame) >> 4) ^ ((j + frame) >> 4)) & 15;
59  py[0] = py[1] = py[vid->nStride] = py[vid->nStride + 1] = 0x80 + z * 0x8;
60  pu[0] = 0x00 + z * 0x10;
61  pv[0] = 0x80 + z * 0x30;
62  py += 2;
63  pu++;
64  pv++;
65  }
66  }
67  *filledLen = (vid->nStride * vid->nSliceHeight * 3) >> 1;
68  return 1;
69 }

91 static int
92 video_encode_test(char *outputfilename)
93 {
94  OMX_VIDEO_PARAM_PORTFORMATTYPE format;
95  OMX_PARAM_PORTDEFINITIONTYPE def;
96  COMPONENT_T *video_encode = NULL;
97  COMPONENT_T *list[5];
98  OMX_BUFFERHEADERTYPE *buf;
99  OMX_BUFFERHEADERTYPE *out;
100  OMX_ERRORTYPE r;
101  ILCLIENT_T *client;
102  int status = 0;
103  int framenumber = 0;
104  FILE *outf;
105 
106  memset(list, 0, sizeof(list));
107 
108  if ((client = ilclient_init()) == NULL) {
109  return -3;
110  }
111 
112  if (OMX_Init() != OMX_ErrorNone) {
113  ilclient_destroy(client);
114  return -4;
115  }
116 
117  // create video_encode
118  r = ilclient_create_component(client, &video_encode, "video_encode",
119  ILCLIENT_DISABLE_ALL_PORTS |
120  ILCLIENT_ENABLE_INPUT_BUFFERS |
121  ILCLIENT_ENABLE_OUTPUT_BUFFERS);
122  if (r != 0) {
123  printf
124  ("ilclient_create_component() for video_encode failed with %x!\n",
125  r);
126  exit(1);
127  }
128  list[0] = video_encode;
129 
130  // get current settings of video_encode component from port 200
131  memset(&def, 0, sizeof(OMX_PARAM_PORTDEFINITIONTYPE));
132  def.nSize = sizeof(OMX_PARAM_PORTDEFINITIONTYPE);
133  def.nVersion.nVersion = OMX_VERSION;
134  def.nPortIndex = 200;
135 
136  if (OMX_GetParameter
137  (ILC_GET_HANDLE(video_encode), OMX_IndexParamPortDefinition,
138  &def) != OMX_ErrorNone) {
139  printf("%s:%d: OMX_GetParameter() for video_encode port 200 failed!\n",
140  __FUNCTION__, __LINE__);
141  exit(1);
142  }
143 
144  print_def(def);
145 
146  // Port 200: in 1/1 115200 16 enabled,not pop.,not cont. 320x240 320x240 @1966080 20
147  def.format.video.nFrameWidth = WIDTH;
148  def.format.video.nFrameHeight = HEIGHT;
149  def.format.video.xFramerate = 30 << 16;
150  def.format.video.nSliceHeight = ALIGN_UP(def.format.video.nFrameHeight, 16);
151  def.format.video.nStride = def.format.video.nFrameWidth;
152  def.format.video.eColorFormat = OMX_COLOR_FormatYUV420PackedPlanar;
153 
154  print_def(def);
155 
156  r = OMX_SetParameter(ILC_GET_HANDLE(video_encode),
157  OMX_IndexParamPortDefinition, &def);
158  if (r != OMX_ErrorNone) {
159  printf
160  ("%s:%d: OMX_SetParameter() for video_encode port 200 failed with %x!\n",
161  __FUNCTION__, __LINE__, r);
162  exit(1);
163  }
164 
165  memset(&format, 0, sizeof(OMX_VIDEO_PARAM_PORTFORMATTYPE));
166  format.nSize = sizeof(OMX_VIDEO_PARAM_PORTFORMATTYPE);
167  format.nVersion.nVersion = OMX_VERSION;
168  format.nPortIndex = 201;
169  format.eCompressionFormat = OMX_VIDEO_CodingAVC;
170 
171  printf("OMX_SetParameter for video_encode:201...\n");
172  r = OMX_SetParameter(ILC_GET_HANDLE(video_encode),
173  OMX_IndexParamVideoPortFormat, &format);
174  if (r != OMX_ErrorNone) {
175  printf
176  ("%s:%d: OMX_SetParameter() for video_encode port 201 failed with %x!\n",
177  __FUNCTION__, __LINE__, r);
178  exit(1);
179  }
180 
181  OMX_VIDEO_PARAM_BITRATETYPE bitrateType;
182  // set current bitrate to 1Mbit
183  memset(&bitrateType, 0, sizeof(OMX_VIDEO_PARAM_BITRATETYPE));
184  bitrateType.nSize = sizeof(OMX_VIDEO_PARAM_BITRATETYPE);
185  bitrateType.nVersion.nVersion = OMX_VERSION;
186  bitrateType.eControlRate = OMX_Video_ControlRateVariable;
187  bitrateType.nTargetBitrate = 1000000;
188  bitrateType.nPortIndex = 201;
189  r = OMX_SetParameter(ILC_GET_HANDLE(video_encode),
190  OMX_IndexParamVideoBitrate, &bitrateType);
191  if (r != OMX_ErrorNone) {
192  printf
193  ("%s:%d: OMX_SetParameter() for bitrate for video_encode port 201 failed with %x!\n",
194  __FUNCTION__, __LINE__, r);
195  exit(1);
196  }
197 
198 
199  // get current bitrate
200  memset(&bitrateType, 0, sizeof(OMX_VIDEO_PARAM_BITRATETYPE));
201  bitrateType.nSize = sizeof(OMX_VIDEO_PARAM_BITRATETYPE);
202  bitrateType.nVersion.nVersion = OMX_VERSION;
203  bitrateType.nPortIndex = 201;
204 
205  if (OMX_GetParameter
206  (ILC_GET_HANDLE(video_encode), OMX_IndexParamVideoBitrate,
207  &bitrateType) != OMX_ErrorNone) {
208  printf("%s:%d: OMX_GetParameter() for video_encode for bitrate port 201 failed!\n",
209  __FUNCTION__, __LINE__);
210  exit(1);
211  }
212  printf("Current Bitrate=%u\n",bitrateType.nTargetBitrate);
213 
214 
215 
216  printf("encode to idle...\n");
217  if (ilclient_change_component_state(video_encode, OMX_StateIdle) == -1) {
218  printf
219  ("%s:%d: ilclient_change_component_state(video_encode, OMX_StateIdle) failed",
220  __FUNCTION__, __LINE__);
221  }
222 
223  printf("enabling port buffers for 200...\n");
224  if (ilclient_enable_port_buffers(video_encode, 200, NULL, NULL, NULL) != 0) {
225  printf("enabling port buffers for 200 failed!\n");
226  exit(1);
227  }
228 
229  printf("enabling port buffers for 201...\n");
230  if (ilclient_enable_port_buffers(video_encode, 201, NULL, NULL, NULL) != 0) {
231  printf("enabling port buffers for 201 failed!\n");
232  exit(1);
233  }
234 
235  printf("encode to executing...\n");
236  ilclient_change_component_state(video_encode, OMX_StateExecuting);
237 
238  outf = fopen(outputfilename, "w");
239  if (outf == NULL) {
240  printf("Failed to open '%s' for writing video\n", outputfilename);
241  exit(1);
242  }
243 
244  printf("looping for buffers...\n");
245  do {
246  buf = ilclient_get_input_buffer(video_encode, 200, 1);
247  if (buf == NULL) {
248  printf("Doh, no buffers for me!\n");
249  } else {
250  /* fill it */
251  generate_test_card(buf->pBuffer, &buf->nFilledLen, framenumber++, &def);
252 
253  if (OMX_EmptyThisBuffer(ILC_GET_HANDLE(video_encode), buf) !=
254  OMX_ErrorNone) {
255  printf("Error emptying buffer!\n");
256  }
257 
258  out = ilclient_get_output_buffer(video_encode, 201, 1);
259 
260  if (out != NULL) {
261  if (out->nFlags & OMX_BUFFERFLAG_CODECCONFIG) {
262  int i;
263  for (i = 0; i < out->nFilledLen; i++)
264  printf("%x ", out->pBuffer[i]);
265  printf("\n");
266  }
267 
268  r = fwrite(out->pBuffer, 1, out->nFilledLen, outf);
269  if (r != out->nFilledLen) {
270  printf("fwrite: Error emptying buffer: %d!\n", r);
271  } else {
272  printf("Writing frame %d/%d, len %u\n", framenumber, NUMFRAMES, out->nFilledLen);
273  }
274  out->nFilledLen = 0;
275  } else {
276  printf("Not getting it :(\n");
277  }
278 
279  r = OMX_FillThisBuffer(ILC_GET_HANDLE(video_encode), out);
280  if (r != OMX_ErrorNone) {
281  printf("Error sending buffer for filling: %x\n", r);
282  }
283  }
284  }
285  while (framenumber < NUMFRAMES);
286 
287  fclose(outf);
288 
289  printf("Teardown.\n");
290 
291  printf("disabling port buffers for 200 and 201...\n");
292  ilclient_disable_port_buffers(video_encode, 200, NULL, NULL, NULL);
293  ilclient_disable_port_buffers(video_encode, 201, NULL, NULL, NULL);
294 
295  ilclient_state_transition(list, OMX_StateIdle);
296  ilclient_state_transition(list, OMX_StateLoaded);
297 
298  ilclient_cleanup_components(list);
299 
300  OMX_Deinit();
301 
302  ilclient_destroy(client);
303  return status;
304 }
