107 static MMAL_STATUS_T aaf_track_create(MMAL_COMPONENT_T *component)
108 {
109  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
110  MMAL_PORT_T *port = component->input[0];
111  audio_channel_mask_t channels_mask;
112  int frame_count = 256;
113 
114  /* Reset track on format change. Can do better than that? */
115  if (module->track != NULL)
116  aaf_track_destroy(component);
117 
118  channels_mask = audio_channel_out_mask_from_count(port->format->es->audio.channels);
119  LOG_INFO("%s(%p) %4.4s, %i Hz, mask %x, chan %i", port->name, port,
120  (char *)&port->format->encoding,
121  (int)port->format->es->audio.sample_rate, channels_mask,
122  (int)port->format->es->audio.channels);
123 
124  AudioTrack::getMinFrameCount(&frame_count);
125  if (port->format->encoding == MMAL_ENCODING_AC3)
126  frame_count = SPDIF_AC3_FRAME_SIZE;
127  else if (port->format->encoding == MMAL_ENCODING_EAC3)
128  frame_count = SPDIF_AC3_FRAME_SIZE * 4;
129  frame_count *= 2; /* Twice the minimum should be enough */
130 
131  module->track = new AudioTrack(AUDIO_STREAM_MUSIC,
132  port->format->es->audio.sample_rate,
133  encoding_to_audio_format(port->format->encoding), channels_mask,
134  frame_count, port->format->encoding == MMAL_ENCODING_PCM_SIGNED ?
135  AUDIO_OUTPUT_FLAG_NONE : AUDIO_OUTPUT_FLAG_DIRECT,
136  &aaf_track_callback, port, 0);
137 
138  if (module->track == NULL || module->track->initCheck() != OK)
139  {
140  LOG_ERROR("%s(%p): track creation failed", port->name, port);
141  module->track = NULL;
142  return MMAL_ENOSYS;
143  }
144 
145  return MMAL_SUCCESS;
146 }

148 static void aaf_track_callback(int event, void *user, void *info)
149 {
150  MMAL_PORT_T *port = (MMAL_PORT_T *)user;
151  MMAL_COMPONENT_T *component = port->component;
152  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
153  MMAL_BUFFER_HEADER_T *buffer;
154  AudioTrack::Buffer *trackbuf = NULL;
155  unsigned int bytes, space;
156  uint8_t *dest;
157 
158  if (event == AudioTrack::EVENT_UNDERRUN)
159  LOG_ERROR("underrun");
160 
161  if (event != AudioTrack::EVENT_MORE_DATA)
162  return;
163 
164  trackbuf = (AudioTrack::Buffer *)info;
165  space = trackbuf->size;
166  dest = (uint8_t *)trackbuf->raw;
167  trackbuf->size = 0;
168 
169  if (!mmal_queue_length(module->queue))
170  {
171  LOG_ERROR("no buffers queued");
172  return;
173  }
174 
175  while (space > 0)
176  {
177  buffer = mmal_queue_get(module->queue);
178  if (!buffer)
179  break;
180 
181  bytes = MMAL_MIN(buffer->length, space);
182  memcpy(dest, buffer->data + buffer->offset, bytes);
183  buffer->offset += bytes;
184  buffer->length -= bytes;
185  dest += bytes;
186  space -= bytes;
187  trackbuf->size += bytes;
188 
189  if (buffer->length)
190  {
191  /* Re-queue */
192  mmal_queue_put_back(module->queue, buffer);
193  continue;
194  }
195 
196  /* Handle the EOS */
197  if (buffer->flags & MMAL_BUFFER_HEADER_FLAG_EOS)
198  mmal_event_eos_send(port);
199 
200  buffer->offset = 0;
201  mmal_port_buffer_header_callback(port, buffer);
202  }
203 
204  module->lock->lock();
205  module->bytes_queued -= trackbuf->size;
206  module->lock->unlock();
207 }

209 static MMAL_STATUS_T aaf_track_state_update(MMAL_COMPONENT_T *component)
210 {
211  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
212  TRACK_STATE_T new_state = TRACK_STATE_STOPPED;
213 
214  if (module->track == NULL)
215  return MMAL_SUCCESS;
216 
217  if (module->is_enabled)
218  {
219  MMAL_RATIONAL_T scale = mmal_port_clock_scale_get(component->clock[0]);
220  new_state = TRACK_STATE_PAUSED;
221  if (scale.den && scale.den == scale.num)
222  new_state = TRACK_STATE_RUNNING;
223  }
224 
225  if (new_state == module->state)
226  return MMAL_SUCCESS; /* Nothing to do */
227 
228  if (module->state == TRACK_STATE_STOPPED && new_state == TRACK_STATE_RUNNING)
229  {
230  module->track->start();
231  }
232  else if (module->state == TRACK_STATE_RUNNING)
233  {
234  if (new_state == TRACK_STATE_STOPPED)
235  module->track->stop();
236  else if (new_state == TRACK_STATE_PAUSED)
237  module->track->pause();
238  }
239  else if (module->state == TRACK_STATE_PAUSED)
240  {
241  if (new_state == TRACK_STATE_STOPPED)
242  module->track->stop();
243  else if (new_state == TRACK_STATE_RUNNING)
244  module->track->start();
245  }
246 
247  module->state = new_state;
248  return MMAL_SUCCESS;
249 }

272 static MMAL_STATUS_T aaf_port_set_format(MMAL_PORT_T *port)
273 {
274  MMAL_COMPONENT_T *component = port->component;
275 
276  if (!mmal_format_compare(port->format, component->priv->module->format))
277  return MMAL_SUCCESS;
278 
279  /* Check the format is supported */
280  if (encoding_to_audio_format(port->format->encoding) == AUDIO_FORMAT_INVALID)
281  {
282  LOG_ERROR("port does not support '%4.4s'", (char *)&port->format->encoding);
283  return MMAL_EINVAL;
284  }
285 
286  /* Specific checks for PCM */
287  if (port->format->encoding == MMAL_ENCODING_PCM_SIGNED)
288  {
289 
290  if (port->format->es->audio.bits_per_sample != 16 &&
291  port->format->es->audio.bits_per_sample != 32)
292  {
293  LOG_ERROR("port does not support '%4.4s' at %ibps",
294  (char *)&port->format->encoding,
295  port->format->es->audio.bits_per_sample);
296  return MMAL_EINVAL;
297  }
298 
299  if (!audio_channel_out_mask_from_count(port->format->es->audio.channels))
300  {
301  LOG_ERROR("%s invalid channels mask from %i", port->name,
302  (int)port->format->es->audio.channels);
303  return MMAL_ENOSYS;
304  }
305  }
306 
307  mmal_format_copy(component->priv->module->format, port->format);
308 
309  return aaf_track_create(component);
310 }

366 static MMAL_STATUS_T aaf_port_send(MMAL_PORT_T *port, MMAL_BUFFER_HEADER_T *buffer)
367 {
368  MMAL_COMPONENT_T *component = port->component;
369  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
370  unsigned int bits_per_sample, channels, sample_rate;
371  uint32_t aaf_bytes_queued = 0;
372  int64_t latency, ts;
373 
374  /* Handle event buffers */
375  if (buffer->cmd)
376  {
377  LOG_ERROR("discarding event %i on port %p", (int)buffer->cmd, port);
378  buffer->length = 0;
379  mmal_port_buffer_header_callback(port, buffer);
380  return MMAL_SUCCESS;
381  }
382 
383  if (module->status != MMAL_SUCCESS)
384  return module->status;
385 
386  bits_per_sample = port->format->es->audio.bits_per_sample;
387  channels = port->format->es->audio.channels;
388  sample_rate = port->format->es->audio.sample_rate;
389 
390  if (port->format->encoding == MMAL_ENCODING_AC3 ||
391  port->format->encoding == MMAL_ENCODING_EAC3)
392  {
393  uint32_t aaf_latency = 0;
394  AudioSystem::getOutputLatency(&aaf_latency, AUDIO_STREAM_MUSIC);
395  latency = aaf_latency * 1000LL;
396 
397  bits_per_sample = 16;
398  channels = 2;
399  if (port->format->encoding == MMAL_ENCODING_EAC3 &&
400  sample_rate <= 48000)
401  sample_rate *= 4;
402  aaf_bytes_queued = module->track->frameCount();
403  }
404  else
405  {
406  latency = module->track->latency() * 1000LL;
407  }
408 
409  /* Keep aaf_track_callback from sending more samples */
410  module->lock->lock();
411 
412  module->bytes_queued += buffer->length;
413  latency += (module->bytes_queued + aaf_bytes_queued) / channels /
414  (bits_per_sample / 8) * 1000000LL / sample_rate;
415  ts = buffer->pts - latency;
416 
417  module->lock->unlock();
418 
419  mmal_port_clock_media_time_set(component->clock[0], ts);
420 
421  mmal_queue_put(module->queue, buffer);
422 
423  return MMAL_SUCCESS;
424 }

447 static MMAL_STATUS_T mmal_component_create_aaf(const char *name, MMAL_COMPONENT_T *component)
448 {
449  MMAL_COMPONENT_MODULE_T *module;
450 
451  /* Check we're the requested component */
452  if(strcmp(name, "aaf." MMAL_AUDIO_RENDER))
453  return MMAL_ENOENT;
454 
455  /* Allocate our module context */
456  component->priv->module = module = (MMAL_COMPONENT_MODULE_T *)vcos_calloc(1, sizeof(*module), "mmal module");
457  if(!module)
458  return MMAL_ENOMEM;
459  module->lock = new Mutex();
460  if (!module->lock)
461  goto error;
462 
463  /* Allocate the ports for this component */
464  component->input = mmal_ports_alloc(component, 1, MMAL_PORT_TYPE_INPUT, 0);
465  if(!component->input)
466  goto error;
467  component->input_num = 1;
468 
469  /* Create the clock port (clock ports are managed by the framework) */
470  component->clock = mmal_ports_clock_alloc(component, 1, 0, aaf_clock_event);
471  if (!component->clock)
472  goto error;
473  component->clock_num = 1;
474 
475  module->queue = mmal_queue_create();
476  if(!module->queue)
477  goto error;
478 
479  module->format = mmal_format_alloc();
480  if(!module->format)
481  goto error;
482 
483  component->input[0]->priv->pf_set_format = aaf_port_set_format;
484  component->input[0]->priv->pf_enable = aaf_port_enable;
485  component->input[0]->priv->pf_disable = aaf_port_disable;
486  component->input[0]->priv->pf_flush = aaf_port_flush;
487  component->input[0]->priv->pf_send = aaf_port_send;
488  component->input[0]->buffer_num_min = INPUT_MIN_BUFFER_NUM;
489  component->input[0]->buffer_num_recommended = INPUT_RECOMMENDED_BUFFER_NUM;
490  component->input[0]->format->type = MMAL_ES_TYPE_AUDIO;
491 
492  component->priv->pf_destroy = aaf_component_destroy;
493  return MMAL_SUCCESS;
494 
495  error:
496  aaf_component_destroy(component);
497  return MMAL_ENOMEM;
