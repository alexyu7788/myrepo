725  // read packets from the file, and stash them away.
726  void read_sequential()
727  {
728  // Use the same buffer for reading each of the packets. It's cheaper to copy the (usually small) data
729  // from this buffer than to create a new one for each packet, then resize it.
730  // 256k is the size used in the other program, and is bigger than any known packet. Increase it
731  // if a bigger one is found.
732  std::vector<uint8_t> buffer(256*1024, 0);
733 
734  std::map<uint32_t, size_t> packet_counts;
735 
736  // Count of memory used for packet buffers.
737  size_t memory_buffered = 0;
738 
739  if (configuration.dump_packets)
740  {
741  *configuration.dump_packets << "pts" << ",\t" << "track" << ",\t" << "size" << ",\t" << "crc" << std::endl;
742  }
743 
744  while (1) // We break out at EOF.
745  {
746  // Packet object
747  PACKET_DATA_T packet;
748 
749  // Use the shared buffer for the moment,
750  packet.info.data = &buffer[0];
751  packet.info.buffer_size = buffer.size();
752 
753  // Read data.
754  status = vc_container_read(p_ctx, &packet.info, 0);
755 
756  if (packet.info.size >= buffer.size())
757  {
758  std::cerr << "Packet size limit exceeeded. Increase internal buffer size!";
759  exit(VC_CONTAINER_ERROR_FAILED);
760  }
761 
762  if (status == VC_CONTAINER_SUCCESS)
763  {
764  // Calculate the CRC of the packet
765  packet.crc = crc32buf(&buffer[0], packet.info.size);
766 
767  // If there is any data, and we haven't exceeded our size limit...
768  if ((packet.info.size > 0) && (memory_buffered < configuration.mem_max))
769  {
770  // Copy the data we read across from the big buffer to our local one
771  packet.buffer.assign(buffer.begin(), buffer.begin() + packet.info.size);
772 
773  // count how much we have buffered
774  memory_buffered += packet.info.size;
775 
776  // wipe the bit of the big buffer we used
777  memset(&buffer[0], 0, packet.info.size);
778  }
779  else
780  {
781  // not storing any data, either because there is none or we have no space.
782  packet.buffer.clear();
783  }
784 
785  // Clear the pointer in our data. It won't be valid later.
786  packet.info.data = 0;
787 
788  // Set the size of the buffer to match what is really there.
789  packet.info.buffer_size = packet.info.size;
790 
791  // count the packet
792  ++packet_counts[packet.info.track];
793 
794  // store it
795  all_packets.push_back(packet);
796 
797  // perhaps dump it
798  if (configuration.dump_packets)
799  {
800  if (packet.info.pts < 0)
801  {
802  *configuration.dump_packets << "-";
803  }
804  else
805  {
806  *configuration.dump_packets << packet.info.pts;
807  }
808 
809  *configuration.dump_packets
810  << ",\t" << packet.info.track << ",\t" << packet.info.size << ",\t" << packet.crc << std::endl;
811  }
812  }
813  else if (status == VC_CONTAINER_ERROR_EOS)
814  {
815  break;
816  }
817  else
818  {
819  error_logger << "error reading file " << configuration.source_name << " Code " << status << std::ends;
820  }
821  }
822 
823  std::cout << std::dec << "File has " << packet_counts.size() << " tracks." << std::endl;
824 
825  // Print the packet count for each stream. Also, while iterating, save all the stream numbers.
826  for(std::map<uint32_t, size_t>::const_iterator stream = packet_counts.begin(); stream != packet_counts.end(); ++stream)
827  {
828  // Print the packet count
829  std::cout << "Stream " << stream->first << " has " << stream->second << " packets." << std::endl;
830 
831  // Note that we have a stream with this number.
832  all_streams.insert(stream->first);
833  };
834 
835  if (p_ctx->tracks_num != packet_counts.size())
836  {
837  error_logger << "The file header claims " << p_ctx->tracks_num
838  << " but " << packet_counts.size() << " streams were found" << std::ends;
839  }

882  // Check which locations can be accessed by a seek, with or without VC_CONTAINER_SEEK_FLAG_FORWARD.
883  void check_indices(VC_CONTAINER_SEEK_FLAGS_T direction)
884  {
885  STREAM_TIMED_PACKETS_T& index_positions = direction == 0 ? reverse_index_positions : forward_index_positions;
886  // Go through each stream that contains key packets (usually only one)
887  for (STREAM_TIMED_PACKETS_T::const_iterator stream_keys = all_key_packets.begin();
888  stream_keys != all_key_packets.end();
889  ++stream_keys)
890  {
891  // Start with a PTS just after the last key packet, and repeat until we fall off the beginning.
892  for (PTS_T target_pts = stream_keys->second.rbegin()->first + 1; target_pts >= 0; /* no decrement */)
893  {
894  // copy the PTS value to pass to vc_container_seek
895  PTS_T actual_pts = target_pts;
896 
897  // Seek to that position in the file.
898  status = vc_container_seek(p_ctx, &actual_pts, VC_CONTAINER_SEEK_MODE_TIME, direction);
899 
900  if (status != VC_CONTAINER_SUCCESS)
901  {
902  error_logger << "Error " << status << " seeking to PTS " << target_pts << std::ends;
903  continue; // if errors are not fatal we'll try again 1uS earlier.
904  }
905 
906  // Check whether this seek reported that it went somewhere sensible
907  check_correct_seek(direction, actual_pts, target_pts);
908 
909  // Validate that the place it said we arrived at is correct - read info for the first packet in any stream
910  {
911  PACKET_DATA_T packet;
912  status = vc_container_read(p_ctx, &packet.info, VC_CONTAINER_READ_FLAG_INFO);
913 
914  if (status != VC_CONTAINER_SUCCESS)
915  {
916  error_logger << "Error " << status << " reading info for packet at PTS " << target_pts << std::ends;
917  continue; // stop trying to read at this PTS.
918  }
919 
920  if (packet.info.pts != actual_pts)
921  {
922  error_logger << "Incorrect seek. Container claimed to have arrived at "
923  << actual_pts << " but first packet ";
924 
925  if (packet.info.pts < 0)
926  {
927  error_logger << " had no PTS";
928  }
929  else
930  {
931  error_logger << "was at " << packet.info.pts;
932  }
933 
934  error_logger << std::ends;
935  }
936  }
937 
938  // We'll perform reads until we've had a packet with PTS from every stream.
939  for (std::set<STREAM_T> unfound_streams = all_streams;
940  !unfound_streams.empty();
941  /* unfound_streams.erase(packet.info.track) */ )
942  {
943  // Read a packet. We can't be sure what track it will be from, so first read the info...
944  PACKET_DATA_T packet;
945  status = vc_container_read(p_ctx, &packet.info, VC_CONTAINER_READ_FLAG_INFO);
946 
947  if (status != VC_CONTAINER_SUCCESS)

950  break; // stop trying to read at this PTS.
951  }
952 
953  // allocate a buffer.
954  packet.buffer.resize(packet.info.size);
955  packet.info.data = &packet.buffer[0];
956  packet.info.buffer_size = packet.info.size;
957 
958  // perform the read
959  status = vc_container_read(p_ctx, &packet.info, 0);
960 
961  if (status != VC_CONTAINER_SUCCESS)
962  {
963  error_logger << "Error " << status << " reading packet following PTS " << target_pts << std::ends;
964  break; // stop trying to read at this PTS.
965  }
966 
967  STREAM_T stream = packet.info.track;
968 
969  // If it has no PTS we can't use it. Read another one.
970  if (packet.info.pts < 0)
971  {
972  // The first packet we found for a stream has no PTS.
973  // That's odd, because we wouldn't be able to play it. However,
974  // if the stream doesn't permit forcing it may be inevitable.
975  if ((unfound_streams.find(stream) != unfound_streams.end())
976  && ((p_ctx->capabilities & VC_CONTAINER_CAPS_FORCE_TRACK) != 0))
977  {
978  error_logger << "Packet in stream " << stream
979  << " has no PTS after seeking to PTS " << target_pts << std::ends;
980  }
981  continue;
982  }
983 
984  // Search for the PTS collection for this stream
985  STREAM_TIMED_PACKETS_T::iterator stream_packets = all_pts_packets.find(stream);
986 
987  if (stream_packets == all_pts_packets.end())
988  {
989  error_logger << "Packet from unknown stream " << stream
990  << "found when reading packet at PTS " << target_pts << std::ends;
991  continue; // try reading another packet.
992  }
993 
994  // Look for the packet for this PTS in this stream.
995  TIMED_PACKETS_T::const_iterator expected_packet = stream_packets->second.find(packet.info.pts);
996 
997  if (expected_packet == stream_packets->second.end())
998  {
999  error_logger << "Read packet from stream " << stream << " has unknown PTS " << packet.info.pts << std::ends;
1000  continue;
1001  }
1002 
1003  // calculate the CRC
1004  packet.crc = crc32buf(&packet.buffer[0], packet.info.size);
1005 
1006  // Validate that the data is the data we found when we did the sequential reads.
1007  std::string anyerror = compare_packets(*expected_packet->second, packet);
1008 
1009  if (!anyerror.empty())
1010  {
1011  error_logger << "Incorrect data found at PTS " << actual_pts << anyerror << std::ends;
1012  }
1013 
1014  // If this is the first packet we found for this stream
1015  if (unfound_streams.find(stream) != unfound_streams.end())
1016  {
1017  // Store the packet we found. Note we key it by the PTS we used for the seek,
1018  // not its own PTS. This should mean next time we seek to that PTS we'll get this packet.
1019  index_positions[stream].insert(std::make_pair(target_pts, expected_packet->second));
1020 
1021  // Check whether the time on the packet is reasonable - it ought to be close to the time achieved.
1022  check_seek_tolerance(packet, actual_pts);
1023 
1024  // Now we've found a packet from this stream we're done with it, so note we don't care about it any more.
1025  unfound_streams.erase(stream);
1026  }
1027  } // repeat until we've had a packet for every track
1028 
1029  // Adjust the target location
1030  if (actual_pts > target_pts)
1031  {
1032  // Find the next packet down from where we looked, and try there.
1033  TIMED_PACKETS_T this_stream_keys = stream_keys->second;
1034  TIMED_PACKETS_T::const_iterator next_packet = this_stream_keys.lower_bound(target_pts);
1035 
1036  // If there is such a packet that it has more than this PTS, and it isn't the first packet
1037  if ((next_packet != this_stream_keys.begin()) &&
1038  (next_packet != this_stream_keys.end()))
1039  {
1040  // pull out the PTS from the one before, and ask for a microsecond past it.
1041  PTS_T new_target_pts = (--next_packet)->first + 1;
1042  if (new_target_pts >= target_pts)
1043  {
1044  --target_pts;
1045  }
1046  else
1047  {
1048  target_pts = new_target_pts;
1049  }
1050  }
1051  else
1052  {
1053  // There's no packet earlier than where we are looking.
1054  // First time try 1 next time try zero.
1055  target_pts = target_pts != 1 ? 1 : 0;
1056  }
1057  }
1058  else if (actual_pts < (target_pts - 1))
1059  {
1060  // The place we arrived at is well before where we wanted.
1061  // Next time go just after where we arrived.
1062  target_pts = actual_pts + 1;
1063  }
1064  else
1065  {
1066  // If the place we asked for is where we arrived, next time try one microsecond down.
1067  --target_pts;
1068  }
1069  }
1070  }

1186  // seek to a random selection of places, and see if we get the correct data
1187  void seek_randomly(VC_CONTAINER_SEEK_FLAGS_T direction)
1188  {
1189  // Depending on whether we are doing forward or reverse seeks pick a collection of places a seek can go to.
1190  const STREAM_TIMED_PACKETS_T& index_positions = direction == 0 ? reverse_index_positions : forward_index_positions;
1191 
1192  // Go through each stream in that collection
1193  for (STREAM_TIMED_PACKETS_T::const_iterator i_all_index_packets = index_positions.begin();
1194  i_all_index_packets != index_positions.end();
1195  ++i_all_index_packets)
1196  {
1197  // These are the index packets in the stream.
1198  const TIMED_PACKETS_T& all_index_packets = i_all_index_packets->second;
1199 
1200  // this is the number of seeks we'll perform.
1201  size_t seek_count = all_index_packets.size();
1202 
1203  // If there are more than 100 locations limit it to 100.
1204  if (seek_count > 100)
1205  {
1206  seek_count = 100;
1207  }
1208 
1209  // We want an unsorted list of PTSes.
1210  std::list<TIMED_PACKETS_T::value_type> selected_index_packets;
1211  {
1212  // Picking 100 at random out of a potentially large collection of timestamps in a set is going
1213  // to be an expensive operation - search by key is really fast, but search by position is not.
1214  // This is an attempt to come up with something reasonably efficient even if the collection is large.
1215  std::vector<PTS_T> all_ptses;
1216 
1217  // Reserve enough memory to hold every PTS. This collection should be several orders of magnitude smaller than the file.
1218  all_ptses.reserve(all_index_packets.size());
1219 
1220  // Copy all the PTS values into the vector
1221  std::transform(
1222  all_index_packets.begin(),
1223  all_index_packets.end(),
1224  std::back_inserter(all_ptses),
1225  [](const TIMED_PACKETS_T::value_type& source){ return source.first; }
1226  );
1227 
1228  // Based roughly on the Knuth shuffle, get a random subset of the PTS to the front of the collection.
1229  std::vector<PTS_T>::iterator some_pts;
1230  size_t count;
1231  for (count = 0, some_pts = all_ptses.begin(); count < seek_count; ++count, ++some_pts)
1232  {
1233  // Swap some random PTS into one of the first seek_count locations.
1234  // Note this may well be another of the first seek_count locations, especially if this is most of them
1235  std::iter_swap(some_pts, (all_ptses.begin() + rand() % seek_count));
1236  }
1237 
1238  // Throw away the ones we don't want
1239  all_ptses.resize(seek_count);
1240 
1241  // Copy the iterators for each packet with a selected PTS into the list.
1242  std::transform(
1243  all_ptses.begin(),
1244  all_ptses.end(),
1245  std::back_inserter(selected_index_packets),
1246  [&all_index_packets](PTS_T time){ return *all_index_packets.find(time); });
1247 
1248  // End scope. That will free the rest of all_ptses.
1249  }
1250 
1251  // Loop through the selected packets.
1252  for (std::list<TIMED_PACKETS_T::value_type>::iterator expected = selected_index_packets.begin();
1253  expected != selected_index_packets.end();
1254  ++expected)
1255  {
1256  // Seek to their position & check we got there
1257  const PACKET_DATA_T& target = *expected->second;
1258  PTS_T target_pts = expected->first;
1259 
1260  status = vc_container_seek(p_ctx, &target_pts, VC_CONTAINER_SEEK_MODE_TIME, direction);
1261  check_correct_seek(direction, target_pts, expected->first);
1262 
1263  // Start by initialising the new packet object to match the old one - it's mostly right.
1264  PACKET_DATA_T found_packet = target;
1265 
1266  // If forcing is supported, read the first packet & check it's OK.
1267  if (p_ctx->capabilities & VC_CONTAINER_CAPS_FORCE_TRACK)
1268  {
1269  // There might not be a buffer if we ran out of memory
1270  found_packet.buffer.resize(found_packet.info.size);
1271 
1272  // Set the address
1273  found_packet.info.data = &found_packet.buffer[0];
1274 
1275  // Perform the read
1276  status = vc_container_read(p_ctx, &found_packet.info, VC_CONTAINER_READ_FLAG_FORCE_TRACK);
1277  }
1278  else
1279  {
1280  // as forcing is not supported repeat reads until a packet is found for this track.
1281  do
1282  {
1283  status = vc_container_read(p_ctx, &found_packet.info, VC_CONTAINER_READ_FLAG_INFO);
1284 
1285  // Give up on any error
1286  if (status != VC_CONTAINER_SUCCESS) break;
