78 /*****************************************************************************/
79 static VC_CONTAINER_STATUS_T binary_writer_control( VC_CONTAINER_T *p_ctx,
80  VC_CONTAINER_CONTROL_T operation, va_list args )
81 {
82  VC_CONTAINER_ES_FORMAT_T *format;
83  VC_CONTAINER_TRACK_T *track;
84  VC_CONTAINER_STATUS_T status;
85 
86  switch(operation)
87  {
88  case VC_CONTAINER_CONTROL_TRACK_ADD:
89  format = (VC_CONTAINER_ES_FORMAT_T *)va_arg( args, VC_CONTAINER_ES_FORMAT_T * );
90 
91  /* Allocate and initialise track data */
92  if(p_ctx->tracks_num >= 1) return VC_CONTAINER_ERROR_OUT_OF_RESOURCES;
93  p_ctx->tracks[p_ctx->tracks_num] = track = vc_container_allocate_track(p_ctx, 0);
94  if(!track) return VC_CONTAINER_ERROR_OUT_OF_MEMORY;
95 
96  if(format->extradata_size)
97  {
98  status = vc_container_track_allocate_extradata( p_ctx, track, format->extradata_size );
99  if(status != VC_CONTAINER_SUCCESS)
100  {
101  vc_container_free_track(p_ctx, track);
102  return status;
103  }
104  WRITE_BYTES(p_ctx, format->extradata, format->extradata_size);
105  }
106 
107  vc_container_format_copy(track->format, format, format->extradata_size);
108  p_ctx->tracks_num++;
109  return VC_CONTAINER_SUCCESS;
110 
111  case VC_CONTAINER_CONTROL_TRACK_ADD_DONE:
112  return VC_CONTAINER_SUCCESS;
113 
114  default: return VC_CONTAINER_ERROR_UNSUPPORTED_OPERATION;
115  }
116 }

118 /*****************************************************************************/
119 VC_CONTAINER_STATUS_T binary_writer_open( VC_CONTAINER_T *p_ctx )
120 {
121  const char *extension = vc_uri_path_extension(p_ctx->priv->uri);
122  VC_CONTAINER_MODULE_T *module = 0;
123  VC_CONTAINER_STATUS_T status = VC_CONTAINER_ERROR_FORMAT_INVALID;
124  unsigned int i;
125 
126  /* Check if the user has specified a container */
127  vc_uri_find_query(p_ctx->priv->uri, 0, "container", &extension);
128 
129  /* Check we're the right writer for this */
130  if(!extension)
131  return VC_CONTAINER_ERROR_FORMAT_NOT_SUPPORTED;
132  for(i = 0; extensions[i]; i++)
133  if(!strcasecmp(extension, extensions[i])) break;
134  if(!extensions[i])
135  return VC_CONTAINER_ERROR_FORMAT_NOT_SUPPORTED;
136 
137  /* Allocate our context */
138  module = malloc(sizeof(*module));
139  if(!module) { status = VC_CONTAINER_ERROR_OUT_OF_MEMORY; goto error; }
140  memset(module, 0, sizeof(*module));
141  p_ctx->priv->module = module;
142  p_ctx->tracks = &module->track;
143 
144  p_ctx->priv->pf_close = binary_writer_close;
145  p_ctx->priv->pf_write = binary_writer_write;
146  p_ctx->priv->pf_control = binary_writer_control;
147  return VC_CONTAINER_SUCCESS;
148 
149  error:
150  LOG_DEBUG(p_ctx, "binary: error opening stream (%i)", status);
151  return status;
152 }
