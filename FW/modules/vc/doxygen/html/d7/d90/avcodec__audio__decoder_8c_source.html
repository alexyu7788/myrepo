100 static MMAL_STATUS_T avcodec_component_destroy(MMAL_COMPONENT_T *component)
101 {

104  if (module->codec_context)

108  if (module->codec_context->codec)

128 static MMAL_STATUS_T avcodec_input_port_set_format(MMAL_PORT_T *port)
129 {
130  MMAL_COMPONENT_T *component = port->component;
131  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
132  enum CodecID codec_id;
133  AVCodec *codec;
134 
135  codec_id = encoding_to_codecid(port->format->encoding);
136  if (codec_id == CODEC_ID_NONE ||
137  !(codec = avcodec_find_decoder(codec_id)))
138  {
139  LOG_ERROR("ffmpeg codec id %d (for %4.4s) not recognized",
140  codec_id, (char *)&port->format->encoding);
141  return MMAL_ENXIO;
142  }
143 
144  module->output_buffer_size = AVCODEC_MAX_AUDIO_FRAME_SIZE;
145  if (module->output_buffer)
146  av_free(module->output_buffer);
147  module->output_buffer = av_malloc(module->output_buffer_size);
148 
149  module->codec_context->sample_rate = port->format->es->audio.sample_rate;
150  module->codec_context->channels = port->format->es->audio.channels;
151  module->codec_context->block_align = port->format->es->audio.block_align;
152  module->codec_context->bit_rate = port->format->bitrate;
153  module->codec_context->bits_per_coded_sample = port->format->es->audio.bits_per_sample;
154  module->codec_context->extradata_size = port->format->extradata_size;
155  module->codec_context->extradata =
156  vcos_calloc(1, port->format->extradata_size + FF_INPUT_BUFFER_PADDING_SIZE,
157  "avcodec extradata");
158  if (module->codec_context->extradata)
159  memcpy(module->codec_context->extradata, port->format->extradata,
160  port->format->extradata_size);
161 
162  if (codec->capabilities & CODEC_CAP_TRUNCATED)
163  module->codec_context->flags |= CODEC_FLAG_TRUNCATED;
164 
165  if (avcodec_open(module->codec_context, codec) < 0)
166  {
167  LOG_ERROR("could not open codec");
168  return MMAL_EIO;
169  }
170 
171  /* Set a default format */
172  if (module->codec_context->sample_fmt == AV_SAMPLE_FMT_NONE)
173  module->codec_context->sample_fmt = AV_SAMPLE_FMT_S16;
174 
175  /* Copy format to output */
176  mmal_format_copy(component->output[0]->format, port->format);
177  LOG_DEBUG("avcodec output format %i", module->codec_context->sample_fmt);
178  component->output[0]->format->encoding = samplefmt_to_encoding(module->codec_context->sample_fmt);
179  component->output[0]->format->es->audio.bits_per_sample =
180  samplefmt_to_sample_size(module->codec_context->sample_fmt);
181 
182  component->output[0]->priv->pf_set_format(component->output[0]);
183  return MMAL_SUCCESS;
184 }

204 
205  port->component->priv->module->output_needs_configuring = 0;
206  mmal_component_action_trigger(port->component);
207 
208  return MMAL_SUCCESS;

220 static MMAL_STATUS_T avcodec_port_flush(MMAL_PORT_T *port)
221 {
222  MMAL_COMPONENT_T *component = port->component;
223  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
224  MMAL_BUFFER_HEADER_T *buffer;
225  MMAL_QUEUE_T *queue;
226 
227  if(port->type == MMAL_PORT_TYPE_OUTPUT)
228  queue = module->queue_out;
229  else if(port->type == MMAL_PORT_TYPE_INPUT)
230  queue = module->queue_in;
231  else
232  return MMAL_EINVAL;
233 
234  /* Flush buffers that our component is holding on to.
235  * If the reading thread is holding onto a buffer it will send it back ASAP as well
236  * so no need to care about that. */
237  while((buffer = mmal_queue_get(queue)))
238  mmal_port_buffer_header_callback(port, buffer);
239 
240  return MMAL_SUCCESS;
241 }

258 static void avcodec_send_event_format_changed(MMAL_COMPONENT_T *component, MMAL_PORT_T *port)
259 {
260  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
261  MMAL_BUFFER_HEADER_T *buffer = NULL;
262  MMAL_EVENT_FORMAT_CHANGED_T *event;
263 
264  /* Get an event buffer */
265  module->status = mmal_port_event_get(port, &buffer, MMAL_EVENT_FORMAT_CHANGED);
266  if (module->status != MMAL_SUCCESS)
267  {
268  LOG_ERROR("unable to get an event buffer");
269  return;
270  }
271  event = mmal_event_format_changed_get(buffer);
272 
273  /* Fill in the new format */
274  mmal_format_copy(event->format, port->format);
275  event->format->es->audio.sample_rate = module->codec_context->sample_rate;
276  event->format->es->audio.channels = module->codec_context->channels;
277  event->format->encoding = samplefmt_to_encoding(module->codec_context->sample_fmt);
278  event->format->es->audio.bits_per_sample = samplefmt_to_sample_size(module->codec_context->sample_fmt);
279 
280  /* Pass on the buffer requirements */
281  event->buffer_num_min = port->buffer_num_min;
282  event->buffer_size_min = port->buffer_size_min;
283  event->buffer_size_recommended = event->buffer_size_min;
284  event->buffer_num_recommended = port->buffer_num_recommended;
285 
286  module->output_needs_configuring = 1;
287  mmal_port_event_send(port, buffer);
288 }

305 /*****************************************************************************/
306 static MMAL_STATUS_T avcodec_send_frame(MMAL_COMPONENT_T *component, MMAL_PORT_T *port)
307 {
308  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
309  MMAL_BUFFER_HEADER_T *out;
310  int size, samples;
311 
312  /* Detect format changes */
313  if (module->codec_context->channels != module->channels ||
314  module->codec_context->sample_rate != module->sample_rate ||
315  module->codec_context->sample_fmt != module->sample_fmt)
316  {
317  avcodec_send_event_format_changed(component, port);
318  return MMAL_EAGAIN;
319  }
320 
321  out = mmal_queue_get(module->queue_out);
322  if (!out)
323  return MMAL_EAGAIN;
324 
325  size = module->output_size;
326  if (size > (int)out->alloc_size)
327  size = out->alloc_size;
328 
329  samples = size / module->channels * 8 / module->bits_per_sample;
330  size = samples * module->channels * module->bits_per_sample / 8;
331  out->length = size;
332  out->pts = module->pts;
333  out->flags = 0;
334  memcpy(out->data, module->output, size);
335  module->output_size -= size;
336  module->output += size;
337 
338  if (module->pts != MMAL_TIME_UNKNOWN)
339  {
340  module->last_pts = module->pts;
341  module->samples_since_last_pts = 0;
342  }
343  module->pts = MMAL_TIME_UNKNOWN;
344  module->samples_since_last_pts += samples;
345 
346  if (out->pts == MMAL_TIME_UNKNOWN)
347  out->pts = module->last_pts + module->samples_since_last_pts * 1000000 / module->sample_rate;
348 
349  out->dts = out->pts;
350  mmal_port_buffer_header_callback(port, out);
351  return MMAL_SUCCESS;
352 }

354 /*****************************************************************************/
355 static MMAL_BOOL_T avaudio_do_processing(MMAL_COMPONENT_T *component)
356 {
357  MMAL_COMPONENT_MODULE_T *module = component->priv->module;
358  MMAL_PORT_T *port_in = component->input[0];
359  MMAL_PORT_T *port_out = component->output[0];
360  MMAL_BUFFER_HEADER_T *in;
361  AVPacket avpkt;
362  int used = 0;
363 
364  if (module->output_needs_configuring)
365  return 0;
366 
367  if (module->output_size &&
368  avcodec_send_frame(component, port_out) != MMAL_SUCCESS)
369  return 0;
370  if (module->output_size)
371  return 1;
372 
373  /* Get input buffer to decode */
374  in = mmal_queue_get(module->queue_in);
375  if (!in)
376  return 0;
377 
378  /* Discard empty buffers. EOS buffers are not discarded since they will be used
379  * to flush the codec. */
380  if (!in->length && !(in->flags & MMAL_BUFFER_HEADER_FLAG_EOS))
381  goto end;
382 
383  /* Avcodec expects padded input data */
384  if (in->length && !in->offset)
385  {
386  if(in->length + FF_INPUT_BUFFER_PADDING_SIZE <= in->alloc_size)
387  memset(in->data + in->length, 0, FF_INPUT_BUFFER_PADDING_SIZE);
388  else
389  LOG_WARN("could not pad buffer"); // Try to decode the data anyway
390  }
391 
392  /* The actual decoding */
393  av_init_packet(&avpkt);
394  avpkt.data = in->length ? in->data + in->offset : 0;
395  avpkt.size = in->length;
396  module->output_size = module->output_buffer_size;
397  module->output = module->output_buffer;
398  used = avcodec_decode_audio3(module->codec_context, (int16_t*)module->output,
399  &module->output_size, &avpkt);
400 
401  /* Check for errors */
402  if (used < 0 || used > (int)in->length)
403  {
404  LOG_ERROR("decoding failed (%i), discarding buffer", used);
405  used = in->length;
406  }
407 
408  module->pts = in->dts;
409  if (module->pts == MMAL_TIME_UNKNOWN)
410  module->pts = in->pts;
411 
412  end:
413  in->offset += used;
414  in->length -= used;
415 
416  if (in->length)
417  {
418  mmal_queue_put_back(module->queue_in, in);
419  return 1;
420  }
421 
422  /* We want to keep the EOS buffer until all the frames have been flushed */
423  if ((in->flags & MMAL_BUFFER_HEADER_FLAG_EOS) &&
424  module->output_size)
425  {
426  mmal_queue_put_back(module->queue_in, in);
427  return 1;
428  }
429 
430  /* Send EOS */
431  if ((in->flags & MMAL_BUFFER_HEADER_FLAG_EOS) &&
432  avcodec_send_eos(component, port_out) != MMAL_SUCCESS)
433  {
434  mmal_queue_put_back(module->queue_in, in);
435  return 0;
436  }
437 
438  in->offset = 0;
439  mmal_port_buffer_header_callback(port_in, in);
440  return 1;
441 }

463 static MMAL_STATUS_T mmal_component_create_avcodec(const char *name, MMAL_COMPONENT_T *component)
464 {
465  MMAL_STATUS_T status = MMAL_ENOMEM;
466  MMAL_COMPONENT_MODULE_T *module;
467 
468  /* Check we're the requested component */
469  if(strcmp(name, "avcodec." MMAL_AUDIO_DECODE))
470  return MMAL_ENOENT;
471 
472  /* Allocate our module context */
473  component->priv->module = module = vcos_calloc(1, sizeof(*module), "mmal module");
474  if(!module)
475  return MMAL_ENOENT;
476 
477  /* Allocate the ports for this component */
478  component->input = mmal_ports_alloc(component, 1, MMAL_PORT_TYPE_INPUT, 0);
479  if(!component->input) goto error;
480  component->input_num = 1;
481  component->output = mmal_ports_alloc(component, 1, MMAL_PORT_TYPE_OUTPUT, 0);
482  if(!component->output) goto error;
483  component->output_num = 1;
484 
485  module->queue_in = mmal_queue_create();
486  if(!module->queue_in) goto error;
487  module->queue_out = mmal_queue_create();
488  if(!module->queue_out) goto error;
489 
490  module->codec_context = avcodec_alloc_context();
491  if(!module->codec_context) goto error;
492 
493  component->input[0]->priv->pf_set_format = avcodec_input_port_set_format;
494  component->input[0]->priv->pf_enable = avcodec_port_enable;
495  component->input[0]->priv->pf_disable = avcodec_port_disable;
496  component->input[0]->priv->pf_flush = avcodec_port_flush;
497  component->input[0]->priv->pf_send = avcodec_port_send;
498  component->input[0]->buffer_num_min = INPUT_MIN_BUFFER_NUM;
499  component->input[0]->buffer_num_recommended = INPUT_RECOMMENDED_BUFFER_NUM;
500  component->input[0]->buffer_size_min = INPUT_MIN_BUFFER_SIZE;
501  component->input[0]->buffer_size_recommended = INPUT_RECOMMENDED_BUFFER_SIZE;
502 
503  component->output[0]->priv->pf_set_format = avcodec_output_port_set_format;
504  component->output[0]->priv->pf_enable = avcodec_port_enable;
505  component->output[0]->priv->pf_disable = avcodec_port_disable;
506  component->output[0]->priv->pf_disable = avcodec_port_flush;
507  component->output[0]->priv->pf_send = avcodec_port_send;
508  component->output[0]->buffer_num_min = OUTPUT_MIN_BUFFER_NUM;
509  component->output[0]->buffer_num_recommended = OUTPUT_RECOMMENDED_BUFFER_NUM;
510  component->output[0]->buffer_size_min = OUTPUT_MIN_BUFFER_SIZE;
511  component->output[0]->buffer_size_recommended = OUTPUT_RECOMMENDED_BUFFER_SIZE;
512 
513  component->output[0]->format->type = MMAL_ES_TYPE_AUDIO;
514  component->output[0]->format->encoding = MMAL_ENCODING_PCM_SIGNED_LE;
515 
516  component->priv->pf_destroy = avcodec_component_destroy;
517 
518  status = mmal_component_action_register(component, avcodec_do_processing_loop);
519  if (status != MMAL_SUCCESS)
520  goto error;
521 
522  return MMAL_SUCCESS;
523 
524  error:
525  avcodec_component_destroy(component);
526  return status;
527 }
