98 /***********************************************************
99  * Name: init_ogl
100  *
101  * Arguments:
102  * CUBE_STATE_T *state - holds OGLES model info
103  *
104  * Description: Sets the display, OpenGL|ES context and screen stuff
105  *
106  * Returns: void
107  *
108  ***********************************************************/
109 static void init_ogl(CUBE_STATE_T *state)
110 {
111  int32_t success = 0;
112  EGLBoolean result;
113  EGLint num_config;
114 
115  static EGL_DISPMANX_WINDOW_T nativewindow;
116 
117  DISPMANX_UPDATE_HANDLE_T dispman_update;
118  VC_RECT_T dst_rect;
119  VC_RECT_T src_rect;
120 
121  static const EGLint attribute_list[] =
122  {
123  EGL_RED_SIZE, 8,
124  EGL_GREEN_SIZE, 8,
125  EGL_BLUE_SIZE, 8,
126  EGL_ALPHA_SIZE, 8,
127  EGL_SURFACE_TYPE, EGL_WINDOW_BIT,
128  EGL_NONE
129  };
130 
131  EGLConfig config;
132 
133  // get an EGL display connection
134  state->display = eglGetDisplay(EGL_DEFAULT_DISPLAY);
135  assert(state->display!=EGL_NO_DISPLAY);
136 
137  // initialize the EGL display connection
138  result = eglInitialize(state->display, NULL, NULL);
139  assert(EGL_FALSE != result);
140 
141  // get an appropriate EGL frame buffer configuration
142  result = eglChooseConfig(state->display, attribute_list, &config, 1, &num_config);
143  assert(EGL_FALSE != result);
144 
145  // create an EGL rendering context
146  state->context = eglCreateContext(state->display, config, EGL_NO_CONTEXT, NULL);
147  assert(state->context!=EGL_NO_CONTEXT);
148 
149  // create an EGL window surface
150  success = graphics_get_display_size(0 /* LCD */, &state->screen_width, &state->screen_height);
151  assert( success >= 0 );
152 
153  dst_rect.x = 0;
154  dst_rect.y = 0;
155  dst_rect.width = state->screen_width;
156  dst_rect.height = state->screen_height;
157 
158  src_rect.x = 0;
159  src_rect.y = 0;
160  src_rect.width = state->screen_width << 16;
161  src_rect.height = state->screen_height << 16;
162 
163  state->dispman_display = vc_dispmanx_display_open( 0 /* LCD */);
164  dispman_update = vc_dispmanx_update_start( 0 );
165 
166  state->dispman_element = vc_dispmanx_element_add ( dispman_update, state->dispman_display,
167  0/*layer*/, &dst_rect, 0/*src*/,
168  &src_rect, DISPMANX_PROTECTION_NONE, 0 /*alpha*/, 0/*clamp*/, 0/*transform*/);
169 
170  nativewindow.element = state->dispman_element;
171  nativewindow.width = state->screen_width;
172  nativewindow.height = state->screen_height;
173  vc_dispmanx_update_submit_sync( dispman_update );
174 
175  state->surface = eglCreateWindowSurface( state->display, config, &nativewindow, NULL );
176  assert(state->surface != EGL_NO_SURFACE);
177 
178  // connect the context to the surface
179  result = eglMakeCurrent(state->display, state->surface, state->surface, state->context);
180  assert(EGL_FALSE != result);
181 
182  // Set background color and clear buffers
183  glClearColor(0.15f, 0.25f, 0.35f, 1.0f);
184 
185  // Enable back face culling.
186  glEnable(GL_CULL_FACE);
187 
188  glMatrixMode(GL_MODELVIEW);
189 }

251 /***********************************************************
252  * Name: update_model
253  *
254  * Arguments:
255  * CUBE_STATE_T *state - holds OGLES model info
256  *
257  * Description: Updates model projection to current position/rotation
258  *
259  * Returns: void
260  *
261  ***********************************************************/
262 static void update_model(CUBE_STATE_T *state)
263 {
264  // update position
265  state->rot_angle_x = inc_and_wrap_angle(state->rot_angle_x, state->rot_angle_x_inc);
266  state->rot_angle_y = inc_and_wrap_angle(state->rot_angle_y, state->rot_angle_y_inc);
267  state->rot_angle_z = inc_and_wrap_angle(state->rot_angle_z, state->rot_angle_z_inc);
268  state->distance = inc_and_clip_distance(state->distance, state->distance_inc);
269 
270  glLoadIdentity();
271  // move camera back to see the cube
272  glTranslatef(0.f, 0.f, -state->distance);
273 
274  // Rotate model to new position
275  glRotatef(state->rot_angle_x, 1.f, 0.f, 0.f);
276  glRotatef(state->rot_angle_y, 0.f, 1.f, 0.f);
277  glRotatef(state->rot_angle_z, 0.f, 0.f, 1.f);
278 }

280 /***********************************************************
281  * Name: inc_and_wrap_angle
282  *
283  * Arguments:
284  * GLfloat angle current angle
285  * GLfloat angle_inc angle increment
286  *
287  * Description: Increments or decrements angle by angle_inc degrees
288  * Wraps to 0 at 360 deg.
289  *
290  * Returns: new value of angle
291  *
292  ***********************************************************/
293 static GLfloat inc_and_wrap_angle(GLfloat angle, GLfloat angle_inc)
294 {
295  angle += angle_inc;
296 
297  if (angle >= 360.0)
298  angle -= 360.f;
299  else if (angle <=0)
300  angle += 360.f;
301 
302  return angle;
303 }

305 /***********************************************************
306  * Name: inc_and_clip_distance
307  *
308  * Arguments:
309  * GLfloat distance current distance
310  * GLfloat distance_inc distance increment
311  *
312  * Description: Increments or decrements distance by distance_inc units
313  * Clips to range
314  *
315  * Returns: new value of angle
316  *
317  ***********************************************************/
318 static GLfloat inc_and_clip_distance(GLfloat distance, GLfloat distance_inc)
319 {
320  distance += distance_inc;
321 
322  if (distance >= 120.0f)
323  distance = 120.f;
324  else if (distance <= 40.0f)
325  distance = 40.0f;
326 
327  return distance;

330 /***********************************************************
331  * Name: redraw_scene
332  *
333  * Arguments:
334  * CUBE_STATE_T *state - holds OGLES model info
335  *
336  * Description: Draws the model and calls eglSwapBuffers
337  * to render to screen
338  *
339  * Returns: void
340  *
341  ***********************************************************/
342 static void redraw_scene(CUBE_STATE_T *state)
343 {
344  // Start with a clear screen
345  glClear( GL_COLOR_BUFFER_BIT );
346 
347  // Draw first (front) face:
348  // Bind texture surface to current vertices
349  glBindTexture(GL_TEXTURE_2D, state->tex[0]);
350 
351  // Need to rotate textures - do this by rotating each cube face
352  glRotatef(270.f, 0.f, 0.f, 1.f ); // front face normal along z axis
353 
354  // draw first 4 vertices
355  glDrawArrays( GL_TRIANGLE_STRIP, 0, 4);
356 
357  // same pattern for other 5 faces - rotation chosen to make image orientation 'nice'
358  glBindTexture(GL_TEXTURE_2D, state->tex[1]);
359  glRotatef(90.f, 0.f, 0.f, 1.f ); // back face normal along z axis
360  glDrawArrays( GL_TRIANGLE_STRIP, 4, 4);
361 
362  glBindTexture(GL_TEXTURE_2D, state->tex[2]);
363  glRotatef(90.f, 1.f, 0.f, 0.f ); // left face normal along x axis
364  glDrawArrays( GL_TRIANGLE_STRIP, 8, 4);
365 
366  glBindTexture(GL_TEXTURE_2D, state->tex[3]);
367  glRotatef(90.f, 1.f, 0.f, 0.f ); // right face normal along x axis
368  glDrawArrays( GL_TRIANGLE_STRIP, 12, 4);
369 
370  glBindTexture(GL_TEXTURE_2D, state->tex[4]);
371  glRotatef(270.f, 0.f, 1.f, 0.f ); // top face normal along y axis
372  glDrawArrays( GL_TRIANGLE_STRIP, 16, 4);
373 
374  glBindTexture(GL_TEXTURE_2D, state->tex[5]);
375  glRotatef(90.f, 0.f, 1.f, 0.f ); // bottom face normal along y axis
376  glDrawArrays( GL_TRIANGLE_STRIP, 20, 4);
377 
378  eglSwapBuffers(state->display, state->surface);
379 }

495 static void exit_func(void)
496 // Function to be passed to atexit().
497 {
498  DISPMANX_UPDATE_HANDLE_T dispman_update;
499  int s;
500  // clear screen
501  glClear( GL_COLOR_BUFFER_BIT );
502  eglSwapBuffers(state->display, state->surface);
503 
504  glDeleteTextures(6, state->tex);
505  eglDestroySurface( state->display, state->surface );
506 
507  dispman_update = vc_dispmanx_update_start( 0 );
508  s = vc_dispmanx_element_remove(dispman_update, state->dispman_element);
509  assert(s == 0);
510  vc_dispmanx_update_submit_sync( dispman_update );
511  s = vc_dispmanx_display_close(state->dispman_display);
512  assert (s == 0);
513 
514  // Release OpenGL resources
515  eglMakeCurrent( state->display, EGL_NO_SURFACE, EGL_NO_SURFACE, EGL_NO_CONTEXT );
516  eglDestroyContext( state->display, state->context );
517  eglTerminate( state->display );
518 
519  // release texture buffers
520  free(state->tex_buf1);
521  free(state->tex_buf2);
522  free(state->tex_buf3);
523 
524  printf("\ncube closed\n");
525 } // exit_func()
