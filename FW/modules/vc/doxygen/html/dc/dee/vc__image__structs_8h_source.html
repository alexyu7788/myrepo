203  struct VC_IMAGE_T {
204 #ifdef __HIGHC__
205  VC_IMAGE_TYPE_T type; /* Metaware will use 16 bits for this enum
206  so use the correct type for debug info */
207 #else
208  unsigned short type; /* should restrict to 16 bits */
209 #endif
210  VC_IMAGE_INFO_T info; /* format-specific info; zero for VC02 behaviour */
211  unsigned short width; /* width in pixels */
212  unsigned short height; /* height in pixels */
213  int pitch; /* pitch of image_data array in bytes */
214  int size; /* number of bytes available in image_data array */
215  void *image_data; /* pixel data */
216  VC_IMAGE_EXTRA_T extra; /* extra data like palette pointer */
217  struct vc_metadata_header_s *metadata; /* metadata header for the image */
218  struct opaque_vc_pool_object_s *pool_object; /* nonNULL if image was allocated from a vc_pool */
219  uint32_t mem_handle; /* the mem handle for relocatable memory storage */
220  int metadata_size; /* size of metadata of each channel in bytes */
221  int channel_offset; /* offset of consecutive channels in bytes */
222  uint32_t video_timestamp;/* 90000 Hz RTP times domain - derived from audio timestamp */
223  uint8_t num_channels; /* number of channels (2 for stereo) */
224  uint8_t current_channel;/* the channel this header is currently pointing to */
225  uint8_t linked_multichann_flag;/* Indicate the header has the linked-multichannel structure*/
226  uint8_t is_channel_linked; /* Track if the above structure is been used to link the header
227  into a linked-mulitchannel image */
228  uint8_t channel_index; /* index of the channel this header represents while
229  it is being linked. */
230  uint8_t _dummy[3]; /* pad struct to 64 bytes */
231  };
